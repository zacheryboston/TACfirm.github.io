1  import java.util.Scanner;
  2  
  3  public class Distances
  4  {
  5     public static void main(String[] args)
  6     {
  7        // Read inputs
  8        Scanner in = new Scanner(System.in);
  9        System.out.print("First latitude in degrees: ");
 10        double lat1 = in.nextDouble();
 11        System.out.print("First longitude: ");
 12        double long1 = in.nextDouble();
 13        System.out.print("Second latitude: ");
 14        double lat2 = in.nextDouble();
 15        System.out.print("Second longitude: ");
 16        double long2 = in.nextDouble();
 17  
 18        // convert to radians
 19        lat1 = lat1 * Math.PI / 180;
 20        long1 = long1 * Math.PI / 180;
 21        lat2 = lat2 * Math.PI / 180;
 22        long2 = long2 * Math.PI / 180;
 23  
 24        double angle = Math.acos(Math.sin(lat1) * Math.sin(lat2)
 25           + Math.cos(lat1) * Math.cos(lat2) * Math.cos(long1 - long2));
 26  
 27        final double RADIUS = 3958.76;
 28  
 29        double distance = RADIUS * angle;
 30        System.out.println("Distance: " + distance);
 31        
 32        System.out.println();
 33     }
 34  }
