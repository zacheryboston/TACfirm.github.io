1  import java.util.Scanner;
  2  
  3  /**
  4     This program justifies paragraphs of text so that all lines
  5     have the same length.
  6  */
  7  public class Justify
  8  {
  9     /**
 10        Gets the starting position of the word following a given word.
 11        @param words a string containing words
 12        @param start the starting position of a word
 13        @return the starting position of the next word, or words.length()
 14        if there is no more word. 
 15     */
 16     public static int nextWord(String words, int start)
 17     {
 18        int i = start;
 19        while (i < words.length() && words.charAt(i) != ' ')
 20           i++;
 21        while (i < words.length() && words.charAt(i) == ' ')
 22           i++;
 23        return i;
 24     }
 25  
 26     /**
 27        Counts the words in a string
 28        @param words a string containing words
 29        @return the number of words in the string
 30     */
 31     public static int countWords(String words)
 32     {
 33        int count = 0;
 34        int next = 0;
 35        while (next < words.length())
 36        {
 37           next = nextWord(words, next);
 38           count++;
 39        }
 40        return count;
 41     }
 42  
 43     /**
 44        Formats a string by adding extra spaces so that the 
 45        string extends to a given margin.
 46        @param line a string
 47        @param margin the desired right margin
 48        @return the formatted line
 49     */
 50     public static String format(String line, int margin)
 51     {
 52        int spacesToAdd = margin - line.length();
 53        if (spacesToAdd <= 0) return line;
 54        int gaps = countWords(line) - 1;
 55        if (gaps <= 0) return line;
 56        int spacesPerGap = spacesToAdd / gaps;
 57        int extraSpaces = spacesToAdd % gaps;
 58        int start = 0;
 59        String result = "";
 60        boolean done = false;
 61        while (!done)
 62        {
 63           int next = nextWord(line, start);
 64           result = result + line.substring(start, next);
 65           if (next == line.length()) // Last word of the line
 66           {
 67              done = true;
 68           }
 69           else
 70           {
 71              for (int i = 1; i <= spacesPerGap; i++)
 72              {
 73                 result = result + " ";
 74              }
 75              if (extraSpaces > 0)
 76              {
 77                 result = result + " ";
 78                 extraSpaces--;
 79              }
 80              start = next;
 81           }
 82        }
 83        return result;
 84     }
 85     
 86     public static void main(String[] args)
 87     {
 88        final int MARGIN = 40;
 89  
 90        Scanner in = new Scanner(System.in);
 91        System.out.println("Enter the words of a paragraph, Q to quit:");
 92        System.out.println("---------+---------+---------+---------+");
 93        String line = in.next();
 94        boolean done = line.equals("Q");
 95        while (!done)
 96        {
 97           String word = in.next();
 98           if (word.equals("Q")) 
 99           {
100              System.out.println(line);
101              done = true;
102           }
103           else 
104           {
105              if (line.length() + 1 + word.length() <= MARGIN)
106              {
107                 line = line + " " + word;
108              }
109              else
110              {
111                 System.out.println(format(line, MARGIN));
112                 line = word;
113              }
114           }
115        }
116        System.out.println("---------+---------+---------+---------+");
117     }
118  }
