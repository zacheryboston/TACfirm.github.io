1  import java.util.Scanner;
  2  
  3  public class ReadTime
  4  {
  5     /**
  6        Prompts a user to enter a value within a given range until the user
  7        provides a valid input.
  8        @param low the smallest allowable input
  9        @param high the largest allowable input
 10        @return the value provided by the user (between low and high, inclusive)
 11     */
 12     public static int readIntBetween(int low, int high)
 13     {
 14        int input;
 15        Scanner in = new Scanner(System.in);
 16        do
 17        {
 18           System.out.print("Enter a value between " + low
 19              + " and " + high + ": ");
 20           input = in.nextInt();
 21        }
 22        while (input < low || input > high);
 23        return input;
 24     }
 25  
 26     public static void main(String[] args)
 27     {
 28        System.out.println("Please enter a time; hours, then minutes.");
 29        int hours = readIntBetween(0, 23);
 30        int minutes = readIntBetween(0, 59);
 31        System.out.printf("You entered %d hours and %d minutes.%n",
 32           hours, minutes);
 33     }
 34  }
