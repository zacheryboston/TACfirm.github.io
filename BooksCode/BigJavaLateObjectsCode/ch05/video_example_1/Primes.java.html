1  import java.util.Scanner;
  2  
  3  // Caution: This program has bugs.
  4  public class Primes
  5  {
  6     /**
  7        Tests if an integer is a prime.
  8        @param n any positive integer
  9        @return true if n is a prime, false otherwise
 10     */
 11     public static boolean isprime(int n)
 12     {
 13        if (n == 1)
 14        {
 15           return false;
 16        }
 17          
 18        if (n == 2)
 19        {
 20           // 2 is a prime
 21           return true;
 22        }
 23        if (n % 2 == 0)
 24        {
 25           // No other even number is a prime
 26           return false;
 27        }
 28  
 29        // Try finding a number that divides n
 30  
 31        int k = 3; // No need to divide by 2 since n is odd
 32        // Only need to try divisors up to sqrt(n)
 33        while (k * k <= n)
 34        {
 35           if (n % k == 0)
 36           {
 37              // n is not a prime since it is divisible by k
 38              return false;
 39           }
 40           // Try next odd number
 41           k = k + 2;
 42        }
 43  
 44        // No divisor found. Therefore, n is a prime
 45        return true;
 46     }
 47  
 48     public static void main(String[] args)
 49     {
 50        System.out.print("Please enter the upper limit: ");
 51        Scanner in = new Scanner(System.in);
 52        int n = in.nextInt();
 53        for (int i = 1; i <= n; i = i + 2)
 54        {
 55           if (isprime(i))
 56           {
 57              System.out.println(i);
 58           }
 59        }
 60     }
