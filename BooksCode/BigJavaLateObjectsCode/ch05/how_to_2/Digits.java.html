  3     /**
  4        Computes the sum of the digits of a number.
  5        @param n an integer >= 0
  6        @return the sum of the digits of n
  7     */
  8     public static int digitSum(int n)
  9     {
 10        if (n == 0) { return 0; } // Special case for terminating the recursion
 11        return digitSum(n / 10) + n % 10; // General case
 12     }
