1  import java.nio.file.Path;
  2  import java.nio.file.Paths;
  3  
  4  public class PathDemo
  5  {
  6     public static void main(String[] args)
  7     {
  8        Path inputPath = Paths.get("input.txt");
  9        System.out.println("Relative path: " + inputPath);
 10        Path absolutePath = inputPath.toAbsolutePath();
 11        System.out.println("Absolute path: " + absolutePath);
 12        Path parent = absolutePath.getParent();
 13        System.out.println("Its parent: " + parent);
 14        Path sibling = parent.resolve("output.txt");
 15        System.out.println("A sibling: " + sibling);
 16        Path fileName = sibling.getFileName();
 17        System.out.println("The sibling's file name: " + fileName);
 18        System.out.print("The sibling's path components: ");
 19        
 20        for (Path p : sibling)
 21        {
 22           System.out.print(p + " ");
 23        }
 24        System.out.println();
 25        
 26        Path changedExtension = sibling.getParent().resolve(
 27           sibling.getFileName().toString().replace(".txt", ".bak"));
 28        System.out.println("Changing the extension: " + changedExtension);
 29     }
 30  }
