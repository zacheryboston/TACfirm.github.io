1  import java.io.IOException;
  2  import java.nio.file.Files;
  3  import java.nio.file.Path;
  4  import java.nio.file.Paths;
  5  
  6  public class CreatingAndDeleting
  7  {
  8     public static void main(String[] args) throws IOException
  9     {
 10        // Creating a directory and a file inside it
 11        Files.createDirectory(Paths.get("subdir"));
 12        Path testPath = Paths.get("subdir/test.txt");
 13        Files.createFile(testPath);
 14        
 15        // Checking that the file exists
 16        System.out.println(testPath + " exists: " + Files.exists(testPath));
 17        // But this one doesn't exist
 18        Path badPath = Paths.get("subdir/bad");
 19        System.out.println(badPath + " exists: " + Files.exists(badPath));
 20  
 21        // Confirming that it's a file and not a directory
 22        System.out.println(testPath + " is a file: " + Files.isRegularFile(testPath));
 23        System.out.println(testPath + " is a directory: " + Files.isDirectory(testPath));
 24  
 25        // Deleting what was created
 26        Files.delete(testPath);
 27        Files.delete(testPath.getParent());
 28  
 29        Path tempFile = Files.createTempFile("bigjava", ".txt");
 30        System.out.println("Created a temporary file " + tempFile);
 31        Path tempDir = Files.createTempDirectory("bigjava");
 32        System.out.println("Created a temporary directory " + tempDir);
 33     }
 34  }
