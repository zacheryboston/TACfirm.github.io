1  import java.io.IOException;
  2  import java.nio.file.Files;
  3  import java.nio.file.Path;
  4  import java.nio.file.Paths;
  5  import java.util.Arrays;
  6  import java.util.Collections;
  7  import java.util.List;
  8  import java.util.stream.Collectors;
  9  import java.util.stream.Stream;
 10  
 11  public class UsefulFileOperations
 12  {
 13     public static void main(String[] args) throws IOException
 14     {
 15        // Getting the size of a file
 16        Path input = Paths.get("UsefulFileOperations.java");
 17        long size = Files.size(input);
 18        System.out.println("Size of " + input + ": " + size + " bytes");
 19  
 20        // Reading all lines
 21        List<String> lines = Files.readAllLines(input);
 22        System.out.println("Line 11: " + lines.get(10));
 23  
 24        // Reading all bytes
 25        byte[] bytes = Files.readAllBytes(input);
 26        byte[] firstTen = Arrays.copyOf(bytes, 10);
 27        System.out.println("First ten bytes: " + Arrays.toString(firstTen));
 28        // Writing lines
 29        Collections.sort(lines);
 30        Path output1 = Paths.get(input + ".sorted");
 31        Files.write(output1, lines);
 32        System.out.println("Sorting the lines of " + input + " yields " + output1); 
 33  
 34        // Writing bytes
 35        for (int i = 0; i < bytes.length; i++)
 36        {
 37           bytes[i] = (byte) (bytes[i] + 3);
 38        }
 39        Path output2 = Paths.get(input + ".caesar");
 40        Files.write(output2, bytes);
 41        System.out.println("The Caesar encryption of " + input + " is in " + output2); 
 42  
 43        // Reading a file into a string
 44        String contents = new String(Files.readAllBytes(input), "UTF-8");
 45        // Writing a string to a file
 46        Path output3 = Paths.get(input + ".lowercase");
 47        contents = contents.toLowerCase();
 48        Files.write(output3, contents.getBytes("UTF-8"));
 49        System.out.println("The lowercase of " + input + " is in " + output3);
 50  
 51        String target = ";";
 52        final int MAX_LINES = 10;
 53        List<String> result = null;
 54        try (Stream<String> lineStream = Files.lines(input))
 55        {
 56           result = lineStream
 57              .filter(s -> s.contains(target))
 58              .limit(MAX_LINES)
 59              .collect(Collectors.toList()); 
 60        }
 61        System.out.println("The first ten lines containing a semicolon: "
 62           + result);
 63  
 64        // Copying and moving
 65        Path output4 = Paths.get(input + ".copied");
 66        if (Files.exists(output4)) { Files.delete(output4); }
 67        Files.copy(input, output4);
 68        Path output5 = Paths.get(output4 + ".moved");
 69        if (Files.exists(output5)) { Files.delete(output5); }
 70        Files.move(output4, output5);
 71        System.out.println(input + " was copied to " + output4 + " which was moved to " + output5);
 72     }
 73  }
