1  public class Loan
  2  {
  3     private double interestRate;
  4     private double balance;
  5     private double interestPaid;
  6  
  7     /**
  8        Constructs a loan for a given amount at a given interest rate.
  9        @param amount the amount being loaned
 10        @param rate the interest rate in percent to be applied at
 11        every payment period.
 12     */
 13     public Loan(double amount, double rate)
 14     {
 15        balance = amount;
 16        interestRate = rate;
 17     }
 18  
 19     /**
 20        Get the total amount of interest paid so far.
 21        @return the interest paid
 22     */
 23     public double getInterestPaid() 
 24     {
 25        return interestPaid;
 26     }
 27  
 28     /**
 29        Get the balance of the principal that still must be repaid.
 30        @return the principal balance
 31     */
 32     public double getBalance()
 33     {
 34        return balance;
 35     }
 36  
 37     /**
 38        Make one payment. 
 39        @param payment the amount of the payment
 40     */
 41     public void makePayment(double payment)
 42     {
 43        double interest = balance * interestRate / 100;
 44        interestPaid = interestPaid + interest;
 45        balance = balance - (payment - interest);
 46     }
 47  }
