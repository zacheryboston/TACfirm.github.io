1  /**
  2     A simulated cash register that tracks the item count and
  3     the total amount due.
  4  */
  5  public class CashRegister
  6  {
  7     private int itemCount;
  8     private double totalPrice;
  9     private double taxableTotal;
 10     private double taxRate;
 11  
 12     /**
 13        Constructs a cash register with cleared item count and total.
 14        @param aTaxRate the tax rate to use with this cash register.
 15     */
 16     public CashRegister(double aTaxRate)
 17     {
 18        itemCount = 0;
 19        totalPrice = 0;
 20        taxableTotal = 0;
 21        taxRate = aTaxRate;
 22     }
 23  
 24     /**
 25        Adds an item to this cash register.
 26        @param price the price of this item
 27        @param taxable true if this item is taxable
 28     */
 29     public void addItem(double price, boolean taxable)
 30     {
 31        itemCount++;
 32        totalPrice = totalPrice + price;
 33        if (taxable) 
 34        {
 35           taxableTotal = taxableTotal + price;
 36        }
 37     }
 38  
 39     /**
 40        Gets the price of all items in the current sale.
 41        @return the total amount
 42     */
 43     public double getTotal()
 44     {
 45        return totalPrice + taxableTotal * taxRate / 100; 
 46     }
 47     
 48     /**
 49        Gets the number of items in the current sale.
 50        @return the item count
 51     */
 52     public int getCount()
 53     {
 54        return itemCount; 
 55     }
 56  
 57     /**
 58        Clears the item count and the total.
 59     */
 60     public void clear()
 61     {
 62        itemCount = 0;
 63        totalPrice = 0;
 64        taxableTotal = 0;
 65     }
 66  }
