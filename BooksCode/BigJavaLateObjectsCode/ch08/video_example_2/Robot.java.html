1  public class Robot
  2  {
  3     private int row;
  4     private int column;
  5     private int direction;
  6     private Maze maze;
  7    
  8     private static final int NORTH = 0;
  9     private static final int EAST = 1;
 10     private static final int SOUTH = 2;
 11     private static final int WEST = 3;
 12     private static final int DIRECTIONS = 4;
 13  
 14     public Robot(Maze aMaze, int startingRow, int startingColumn)
 15     {
 16        maze = aMaze;
 17        row = startingRow;
 18        column = startingColumn;
 19        direction = NORTH;
 20     }
 21  
 22     private static int nextRow(int aRow, int aDirection)
 23     {
 24        if (aDirection == NORTH) { return aRow - 1; }
 25        else if (aDirection == SOUTH) { return aRow + 1; }
 26        else { return aRow; }
 27     }
 28  
 29     private static int nextColumn(int aColumn, int aDirection)
 30     {
 31        if (aDirection == WEST) { return aColumn - 1; }
 32        else if (aDirection == EAST) { return aColumn + 1; }
 33        else { return aColumn; }
 34     }
 35     
 36     public void move()
 37     {
 38        if (direction == NORTH) { row--; }
 39        else if (direction == SOUTH) { row++; }
 40        if (direction == WEST) { column--; }
 41        else if (direction == EAST) { column++; }
 42     }
 43  
 44     public void turnRight()
 45     {
 46        direction = (direction + 1) % DIRECTIONS;
 47     }
 48  
 49     public void turnLeft()
 50     {
 51        turnRight();
 52        turnRight();
 53        turnRight();
 54     }
 55  
 56     public boolean seesWall()
 57     {
 58        int rowAhead = row;
 59        int columnAhead = column;
 60        if (direction == NORTH) { rowAhead--; }
 61        else if (direction == SOUTH) { rowAhead++; }
 62        if (direction == WEST) { columnAhead--; }
 63        else if (direction == EAST) { columnAhead++; }
 64        return maze.isWall(rowAhead, columnAhead);
 65     }
 66  
 67     public boolean seesExit()
 68     {
 69        return maze.isExit(row, column);
 70     }
 71  
 72     public String toString()
 73     {
 74        return "Robot[row=" + row + ",column=" + column 
 75           + ",direction=" + "NESW".charAt(direction) + "]";
 76     }   
 77  }
