1  import java.util.ArrayList;
  2  
  3  /**
  4     A maze has its walls marked by * characters and corridors 
  5     by spaces. The maze is assumed not to have
  6     any cycles (i.e., corridors returning to their own start.)
  7  */
  8  public class Maze
  9  {
 10     private String[] cells;
 11  
 12     /**
 13        Constructs a maze from a string describing its contents.
 14        @param contents strings consisting of * and spaces
 15     */
 16     public Maze(String[] contents)
 17     {
 18        cells = contents;
 19     }
 20  
 21     /**
 22        Checks whether a position is an exit.
 23        @param row the row of the position
 24        @param column the column of the position
 25        @return true if the position is an exit
 26     */
 27     public boolean isExit(int row, int column)
 28     {
 29        return (row == 0 || row == cells.length - 1 
 30           || column == 0 || column == cells[row].length() - 1)
 31           && cells[row].charAt(column) == ' ';
 32     }
 33  
 34     /**
 35        Checks whether a position is within the maze and not a wall.
 36        @param row the row of the position
 37        @param column the column of the position
 38        @return true if the position is valid
 39     */
 40     public boolean isWall(int row, int column)
 41     {
 42        return cells[row].charAt(column) == '*';
 43     }
 44  }
