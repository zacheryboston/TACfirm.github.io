1  import java.awt.Color;
  2  import java.awt.Graphics;
  3  import javax.swing.JComponent;
  4  
  5  /**
  6     A component that shows two Italian flags.
  7  */
  8  public class ItalianFlagComponent extends JComponent
  9  {
 10     public void paintComponent(Graphics g)
 11     {
 12        drawItalianFlag(g, 10, 10, 100);
 13        drawItalianFlag(g, 10, 125, 150);
 14     }
 15  
 16     /**
 17        Draws an Italian flag at the given location and size
 18        @param g the graphics object
 19        @param xLeft the x coordinate of the top left corner
 20        @param yTop the y coordinate of the top left corner
 21        @param width the width of the flag
 22     */
 23     void drawItalianFlag(Graphics g, int xLeft, int yTop, int width)
 24     {
 25        g.setColor(Color.GREEN); 
 26        g.fillRect(xLeft, yTop, width / 3, width * 2 / 3);
 27        g.setColor(Color.RED);
 28        g.fillRect(xLeft + 2 * width / 3, yTop, width / 3, width * 2 / 3);
 29        g.setColor(Color.BLACK);
 30        g.drawLine(xLeft + width / 3, yTop, xLeft + width * 2 / 3, yTop);
 31        g.drawLine(xLeft + width / 3, yTop + width * 2 / 3, 
 32           xLeft + width * 2 / 3, yTop + width * 2 / 3);
 33     }
 34  }
