1  import java.awt.event.ActionEvent;
  2  import java.awt.event.ActionListener;
  3  import java.io.File;
  4  import java.io.FileNotFoundException;
  5  import java.util.ArrayList;
  6  import java.util.Scanner;
  7  import javax.swing.JButton;
  8  import javax.swing.JFrame;
  9  import javax.swing.JOptionPane;
 10  import javax.swing.JPanel;
 11  import javax.swing.JScrollPane;
 12  import javax.swing.JTextField;
 13  import javax.swing.JTextArea;
 14  
 15  public class CrossWordFrame extends JFrame
 16  {
 17     private ArrayList<JTextField> letters;
 18     private ArrayList<String> words;
 19     private JPanel letterPanel;
 20     private JPanel mainPanel;
 21     private JButton moreButton;
 22     private JButton fewerButton;
 23     private JButton searchButton;
 24     private JTextArea wordArea;
 25     private static final int INITIAL_LETTERS = 5;
 26     private static final int FRAME_WIDTH = 400;
 27     private static final int FRAME_HEIGHT = 250;
 28  
 29     public CrossWordFrame()
 30     {
 31        mainPanel = new JPanel();
 32        createLetters();
 33        createButtons();
 34        wordArea = new JTextArea(20, 20);
 35        mainPanel.add(new JScrollPane(wordArea));
 36        add(mainPanel);
 37        setSize(FRAME_WIDTH, FRAME_HEIGHT);
 38        readWords();
 39     }
 40  
 41     class MoreButtonListener implements ActionListener
 42     {
 43        public void actionPerformed(ActionEvent event)
 44        {
 45           JTextField letter = new JTextField(1);
 46           letters.add(letter);
 47           letterPanel.add(letter);         
 48           revalidate();
 49        }
 50     }
 51  
 52     class FewerButtonListener implements ActionListener
 53     {
 54        public void actionPerformed(ActionEvent event)
 55        {
 56           if (letters.size() > 0)
 57           {
 58              JTextField letter = letters.remove(letters.size() - 1);
 59              letterPanel.remove(letter);         
 60              revalidate();
 61           }
 62        }
 63     }
 64  
 65     class SearchButtonListener implements ActionListener
 66     {
 67        public void actionPerformed(ActionEvent event)
 68        {
 69           String pattern = "";
 70           for (JTextField letter : letters)
 71           {
 72              String text = letter.getText();
 73              if (text.length() == 0) { pattern = pattern + "?"; }
 74              else { pattern = pattern + text.substring(0, 1); }
 75           }
 76           wordArea.setText("");
 77           for (String word : words)
 78           {
 79              if (matches(pattern, word))
 80              {
 81                 wordArea.append(word + "\n");
 82              }
 83           }
 84        }
 85  
 86        private boolean matches(String pattern, String word)
 87        {
 88           if (pattern.length() != word.length()) { return false; }
 89           for (int i = 0; i < pattern.length(); i++)
 90           {
 91              char p = pattern.charAt(i);
 92              char w = word.charAt(i);
 93              if (p != w && p != '?') { return false; }
 94           }
 95           return true;
 96        }
 97     }
 98  
 99     private void createLetters()
100     {
101        letters = new ArrayList<JTextField>();
102        letterPanel = new JPanel();
103        for (int i = 1; i <= INITIAL_LETTERS; i++)
104        {
105           JTextField letter = new JTextField(1);
106           letters.add(letter);
107           letterPanel.add(letter);
108        }
109        mainPanel.add(letterPanel);
110     }
111  
112     private void createButtons()
113     {
114        moreButton = new JButton("+");
115        ActionListener listener = new MoreButtonListener();
116        moreButton.addActionListener(listener);
117        mainPanel.add(moreButton);
118        fewerButton = new JButton("-");
119        listener = new FewerButtonListener();
120        fewerButton.addActionListener(listener);
121        mainPanel.add(fewerButton);
122        searchButton = new JButton("?");
123        listener = new SearchButtonListener();
124        searchButton.addActionListener(listener);
125        mainPanel.add(searchButton);
126     }
127  
128     private void readWords()
129     {
130        try
131        {
132           words = new ArrayList<String>();
133           Scanner in = new Scanner(new File("words"));
134           while (in.hasNext()) { words.add(in.next()); }
135           in.close();
136        }
137        catch (FileNotFoundException ex)
138        {
139           JOptionPane.showMessageDialog(null, "words file not found");
140        }     
141     }
142  }
