1  import java.util.Scanner;
  2  import java.io.File;
  3  import java.io.IOException;
  4  
  5  public class CatchingExceptions
  6  {
  7     public static void main(String[] args)
  8     {
  9        try
 10        {
 11           String filename = "test1.txt";
 12           Scanner in = new Scanner(new File(filename));
 13           int sum = 0;
 14           while (in.hasNext())
 15           {
 16              String input = in.next();
 17              int value = Integer.parseInt(input);
 18              sum = sum + value;
 19           }
 20           System.out.println("Sum: " + sum);
 21        }
 22        catch (IOException exception)
 23        {
 24           exception.printStackTrace();
 25        }
 26        catch (NumberFormatException exception)
 27        {
 28           System.out.println(exception.getMessage());
 29        }
 30     }
 31  }
