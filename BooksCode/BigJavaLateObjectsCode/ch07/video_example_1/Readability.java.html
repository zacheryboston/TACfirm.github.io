1   import java.io.File;
  2  import java.io.FileNotFoundException;
  3  import java.util.Scanner;
  4  
  5  public class Readability
  6  {
  7     public static void main(String[] args)
  8     {
  9        String filename;
 10        if (args.length == 0)
 11        {
 12           Scanner console = new Scanner(System.in);
 13           System.out.print("Input file: ");
 14           filename = console.nextLine();
 15        }
 16        else
 17        {
 18           filename = args[0];
 19        }
 20        try
 21        {
 22  
 23           Scanner in = new Scanner(new File(filename));
 24           readabilityIndex(in);
 25        }
 26        catch (FileNotFoundException ex)
 27        {
 28           System.out.println("File not found: " + filename);
 29        }
 30     }
 31  
 32     /**
 33        Checks whether a character is a vowel.
 34        @param ch a string of length 1
 35        @return true if ch is a vowel
 36     */
 37     public static boolean isVowel(String ch)
 38     {
 39        return "aeiouy".contains(ch.toLowerCase());
 40     }
 41  
 42     /**
 43        Checks whether a character is a punctuation mark denoting the end of
 44        a sentence.
 45        @param ch a string of length 1
 46        @return true if ch ends a sentence
 47     */
 48     public static boolean isSentenceEnd(String ch)
 49     {
 50        return ".:;?!".contains(ch);
 51     }
 52  
 53     /**
 54        Computes and displays the readability index of a file.
 55        @param in a scanner for the file
 56     */
 57     public static void readabilityIndex(Scanner in)
 58     {
 59        int words = 0;
 60        int syllables = 0;
 61        int sentences = 0;
 62        while (in.hasNext())
 63        {
 64           String word = in.next();
 65           words++;
 66           syllables = syllables + countSyllables(word);
 67           if (isSentenceEnd(word.substring(word.length() - 1)))
 68           { 
 69              sentences++; 
 70           }
 71        }
 72        double index = 0;
 73        if (sentences > 0)
 74        {
 75           index = 206.835 - 84.6 * syllables / words 
 76              - 1.015 * words / sentences;
 77        }
 78        System.out.println("Words: " + words);
 79        System.out.println("Syllables: " + syllables);
 80        System.out.println("Sentences: " + sentences);
 81        System.out.println("Index: " + index);
 82     }
 83  
 84     /**
 85        Counts the syllables in a word.
 86        @param word a word
 87        @return the number of syllables in the word
 88     */
 89     public static int countSyllables(String word)
 90     {
 91        int count = 0;
 92        boolean inVowelGroup = false;
 93        for (int i = 0; i < word.length(); i++)
 94        {
 95           String letter = word.substring(i, i + 1);
 96           if (isVowel(letter))
 97           {
 98              if (!inVowelGroup)
 99              {
100                 inVowelGroup = true;
101                 count++;
102              }
103           }
104           else
105           {
106              inVowelGroup = false;
107           }
108        }
109        if (word.endsWith("e")) { count--; }
110        if (count == 0) { count = 1; }
111        return count;
112     }
