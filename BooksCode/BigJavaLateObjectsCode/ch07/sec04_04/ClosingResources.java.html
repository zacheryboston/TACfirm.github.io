1  import java.io.File;
  2  import java.io.FileNotFoundException;
  3  import java.io.IOException;
  4  import java.io.PrintWriter;
  5  import java.util.Scanner;
  6  
  7  public class ClosingResources
  8  {
  9     public static void main(String[] args) throws FileNotFoundException
 10     {
 11        File inFile = new File("test1.txt");
 12        File outFile = new File("output.txt");
 13        try (Scanner in = new Scanner(inFile); PrintWriter out = new PrintWriter(outFile))
 14        {
 15           while (in.hasNextLine())
 16           {
 17              String result = process(in.nextLine());
 18              out.println(result);
 19           }
 20        } // Both in.close() and out.close() are called here
 21     }
 22  
 23     public static String process(String line)
 24     {
 25        int value = Integer.parseInt(line);
 26        value++;
 27        return "" + value;
 28     }
 29  }
