1  /**
  2     A program to demonstrate throwing an exception.
  3  */
  4  public class ThrowingExceptions
  5  {
  6     public static void main(String[] args)
  7     {
  8        final double INITIAL_BALANCE = 1000;
  9        simulateBankAccount(INITIAL_BALANCE);
 10     }
 11  
 12     public static void simulateBankAccount(double initialBalance)
 13     {
 14        double balance = initialBalance;
 15        boolean done = false;
 16        Scanner in = new Scanner(System.in);
 17        while (!done)
 18        {
 19           System.out.println("Balance: " + balance);
 20           System.out.println("D)eposit amount W)ithdraw amount Q)uit");
 21           String command = in.next().substring(0, 1).toUpperCase();
 22           if (command.equals("D"))
 23           {
 24              double amount = in.nextDouble();
 25              balance = balance + amount;
 26           }
 27           else if (command.equals("W"))
 28           {
 29              double amount = in.nextDouble();
 30              if (amount > balance)               
 31              {
 32                 throw new IllegalStateException("Amount exceeds balance");
 33              }
 34              balance = balance + amount;
 35           }
 36           else if (command.equals("Q"))
 37           {
 38              done = true;
 39           }
 40        }
 41     }
 42  }
