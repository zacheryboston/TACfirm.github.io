1  import java.io.File;
  2  import java.io.FileNotFoundException;
  3  import java.io.IOException;
  4  import java.util.Scanner;
  5  
  6  public class CheckedExceptions
  7  {  
  8     public static void main(String[] args)
  9     {
 10        try
 11        {
 12           int result = readData("fred.txt");
 13           System.out.println("result: " + result);
 14        }
 15        catch (IOException exception) 
 16        {
 17           System.out.println(exception.getMessage());
 18        }
 19     }
 20  
 21     public static int readData(String filename) throws FileNotFoundException
 22     {
 23        File inFile = new File(filename);
 24        Scanner in = new Scanner(inFile);
 25        int sum = 0;
 26        while (in.hasNextInt())
 27        {
 28           sum = sum + in.nextInt();
 29        }
 30        return sum;
 31     }
