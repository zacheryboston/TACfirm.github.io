1  import java.io.File;
  2  import java.io.FileNotFoundException;
  3  import java.util.ArrayList;
  4  import java.util.Scanner;
  5  
  6  public class DigitDistribution
  7  {
  8     /**
  9        Reads all numbers in the input.
 10        @param in a scanner
 11        @return an array list of all numbers in the input
 12     */
 13     public static ArrayList<Double> getNumbers(Scanner in)
 14     {
 15        in.useDelimiter("[^0-9.,]+");
 16        
 17        ArrayList<Double> result = new ArrayList<Double>();
 18        
 19        while (in.hasNext())
 20        {
 21           String next = in.next();
 22           next.replace(",", "");
 23           try
 24           {
 25                 double value = Double.parseDouble(next);
 26                 result.add(value);
 27           }
 28           catch (NumberFormatException ex)
 29           {
 30              // Couldn't convert
 31           }
 32        }
 33        return result;
 34     }
 35  
 36     /**
 37        Gets the counts of the first non-zero digits in a list of numbers
 38        @param values a list of numbers
 39        @param an array of counts whose ith element is the 
 40        count of values starting with the digit i
 41     */
 42     public static int[] getFirstDigitCounts(ArrayList<Double> values)
 43     {
 44        int[] counters = new int[10];
 45        for (double value : values)
 46        {
 47           if (value >= 1) 
 48           {
 49              int firstDigit = ("" + value).charAt(0) - '0';
 50              counters[firstDigit]++;
 51           }
 52        }      
 53        return counters;
 54     }
 55  
 56     public static void main(String[] args)
 57     {
 58        try
 59        {
 60           System.out.print("Input file: ");         
 61           Scanner in = new Scanner(System.in); 
 62           String filename = in.nextLine();
 63           Scanner infile = new Scanner(new File(filename));
 64        
 65           ArrayList<Double> numbers = getNumbers(infile);
 66  
 67           int[] counts = getFirstDigitCounts(numbers);
 68           
 69           for (int i = 1; i <= 9; i++)
 70           {
 71              System.out.println(i + ": " + counts[i]);
 72           }
 73        }
 74        catch (FileNotFoundException ex)
 75        {
 76           System.out.println(ex.getMessage());
 77        }
 78     }
 79  }
