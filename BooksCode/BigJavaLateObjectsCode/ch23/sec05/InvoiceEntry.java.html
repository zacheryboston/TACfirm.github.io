 10  /**
 11     Enters an invoice into the database.
 12     Be sure to add Customer.sql, Product.sql, Invoice.sql, and LineItem.sql
 13     to the database before running this program.
 14  */
 15  public class InvoiceEntry
 16  {
 17     public static void main(String args[])         
 18     {   
 19        if (args.length == 0)
 20        {   
 21           System.out.println(
 22                 "Usage: java -classpath driver_class_path"
 23                 + File.pathSeparator 
 24                 + ". InvoiceEntry propertiesFile");
 25           return;
 26        }
 27  
 28        try 
 29        {
 30           SimpleDataSource.init(args[0]);
 31           try (Connection conn = SimpleDataSource.getConnection();
 32              Scanner in = new Scanner(System.in))
 33           {
 34              addInvoice(in, conn);
 35           }
 36        }
 37        catch (SQLException ex)
 38        {
 39           System.out.println("Database error");
 40           ex.printStackTrace();
 41        }
 42        catch (ClassNotFoundException ex)
 43        {
 44           System.out.println("Error loading database driver");
 45           ex.printStackTrace();
 46        }
 47        catch (IOException ex)
 48        {
 49           System.out.println("Error loading database properties");
 50           ex.printStackTrace();
 51        }
 52     }

 54     public static void addInvoice(Scanner in, Connection conn)
 55        throws SQLException
 56     {      
 57        int customerNumber = newCustomer(conn, in);
 58  
 59        int id = getNewId(conn, "Invoice");
 60        try (PreparedStatement stat = conn.prepareStatement(
 61           "INSERT INTO Invoice VALUES (?, ?, 0)"))
 62        {
 63           stat.setInt(1, id);
 64           stat.setInt(2, customerNumber);
 65           stat.executeUpdate();
 66        }
 67  
 68        boolean done = false;
 69        while (!done)
 70        {
 71           String productCode = nextLine(in, "Product code (D=Done, L=List)");
 72           if (productCode.equals("D")) { done = true; }
 73           else if (productCode.equals("L")) { listProducts(conn); }               
 74           else if (findProduct(conn, productCode))
 75           {
 76              int quantity = nextInt(in, "Quantity");
 77              addLineItem(conn, id, productCode, quantity);
 78           }
 79           else { System.out.println("Invalid product code."); }
 80        }
 81        showInvoice(conn, id);
 82     }

 84     /**
 85        Prompts the user for the customer information and creates a new customer.
 86        @param conn the database connection
 87        @param in the scanner
 88        @return the ID of the new customer
 89     */
 90     private static int newCustomer(Connection conn, Scanner in)
 91        throws SQLException
 92     {
 93        String name = nextLine(in, "Name");
 94        String address = nextLine(in, "Street address");
 95        String city = nextLine(in, "City");
 96        String state = nextLine(in, "State");
 97        String zip = nextLine(in, "Zip");
 98        int id = getNewId(conn, "Customer");
 99        try (PreparedStatement stat = conn.prepareStatement(
100           "INSERT INTO Customer VALUES (?, ?, ?, ?, ?, ?)"))
101        {
102           stat.setInt(1, id);
103           stat.setString(2, name);
104           stat.setString(3, address);
105           stat.setString(4, city);
106           stat.setString(5, state);
107           stat.setString(6, zip);
108           stat.executeUpdate();
109        }
110        return id;
111     }

113     /**
114        Finds a product in the database.
115        @param conn the database connection
116        @param code the product code to search
117        @return true if there is a product with the given code
118     */
119     private static boolean findProduct(Connection conn, String code)
120        throws SQLException
121     {
122        boolean found = false;
123        try (PreparedStatement stat = conn.prepareStatement(
124           "SELECT * FROM Product WHERE Product_Code = ?"))
125        {
126           stat.setString(1, code);
127           ResultSet result = stat.executeQuery();
128           found = result.next();
129        }
130        return found;
131     }

136        @param id the invoice ID
137        @param code the product code
138        @param quantity the quantity to order

140     private static void addLineItem(Connection conn, int id,
141        String code, int quantity) throws SQLException

143        try (PreparedStatement stat = conn.prepareStatement(
144           "INSERT INTO LineItem VALUES (?, ?, ?)"))

146           stat.setInt(1, id);
147           stat.setString(2, code);
148           stat.setInt(3, quantity);
149           stat.executeUpdate();

153     /**
154        Lists all products in the database.
155        @param conn the database connection
156     */
157     private static void listProducts(Connection conn)
158        throws SQLException
159     {
160        try (Statement stat = conn.createStatement())
161        {
162           ResultSet result = stat.executeQuery(
163              "SELECT Product_Code, Description FROM Product");
164           while (result.next())
165           {
166              String code = result.getString(1);
167              String description = result.getString(2);
168              System.out.println(code + " " + description);
169           }
170        }      
171     }

173     /**
174        Gets a new ID for a table. This method should be called from
175        inside a transaction that also creates the new row with this ID.
176        The ID field should have name table_Number and type INTEGER.
177        @param table the table name
178        @return a new ID that has not yet been used.
179     */
180     private static int getNewId(Connection conn, String table)
181        throws SQLException
182     {
183        int max = -1;
184        try (Statement stat = conn.createStatement())
185        {
186           ResultSet result = stat.executeQuery(
187              "SELECT max(" + table + "_Number) FROM " + table);
188           result.next();
189           max = result.getInt(1) + 1;
190        }
191        return max;
192     }

194     /**
195        Shows an invoice.
196        @param conn the database connection
197        @param id the invoice ID
198     */
199     private static void showInvoice(Connection conn, int id)
200        throws SQLException
201     {
202        try (PreparedStatement stat = conn.prepareStatement(
203              "SELECT Customer.Name, Customer.Address, "
204              + "Customer.City, Customer.State, Customer.Zip "
205              + "FROM Customer, Invoice "
206              + "WHERE Customer.Customer_Number = Invoice.Customer_Number "
207              + "AND Invoice.Invoice_Number = ?"))
208        {
209           stat.setInt(1, id);
210           ResultSet result = stat.executeQuery();
211           result.next();
212           System.out.println(result.getString(1));
213           System.out.println(result.getString(2));        
214           System.out.println(result.getString(3).trim() + ", "
215              + result.getString(4) + " " + result.getString(5));
216        }
217                   
218        try(PreparedStatement stat = conn.prepareStatement(
219              "SELECT Product.Product_Code, Product.Description, LineItem.Quantity "
220              + "FROM Product, LineItem "
221              + "WHERE Product.Product_Code = LineItem.Product_Code "
222              + "AND LineItem.Invoice_Number = ?"))
223        {
224           stat.setInt(1, id);
225  
226           ResultSet result = stat.executeQuery();
227           while (result.next())
228           {  
229              String code = result.getString(1);
230              String description = result.getString(2).trim();
231              int qty = result.getInt(3);
232  
233              System.out.println(qty + " x " + code + " " + description);
234           }
235        }
236     }

238     /**
239        Prompts the user and reads a line from a scanner.
240        @param in the scanner
241        @param prompt the prompt
242        @return the string that the user entered
243     */
244     private static String nextLine(Scanner in, String prompt)
245     {
246        System.out.print(prompt + ": ");
247        return in.nextLine();
248     }

250     /**
251        Prompts the user and reads an integer from a scanner.
252        @param in the scanner
253        @param prompt the prompt
254        @return the integer that the user entered
255     */
256     private static int nextInt(Scanner in, String prompt)
257     {
258        System.out.print(prompt + ": ");
259        int result = in.nextInt();
260        in.nextLine(); // Consume newline
261        return result;
262     }
