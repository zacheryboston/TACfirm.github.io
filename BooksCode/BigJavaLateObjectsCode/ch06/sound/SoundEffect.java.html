1  import java.awt.Color;
  2  import java.awt.Graphics;
  3  import java.io.RandomAccessFile;
  4  import java.io.IOException;

  6  import javax.swing.JComponent;
  7  import javax.swing.JFileChooser;
  8  import javax.swing.JFrame;
  9  import javax.swing.JOptionPane;
 10  import javax.sound.sampled.AudioSystem;
 11  import javax.sound.sampled.Clip;

 13  /**
 14     This program processes a 16 bit uncompressed mono .WAV file.
 15  */
 16  public class SoundEffect
 17  {
 18     /**
 19        Processes the sound samples. Modify this function to change
 20        the way the sound is processed.
 21        @param samples the sound samples in the sound file
 22        @param size the number of samples
 23        @param sampleRate the number of samples per second
 24     */
 25     public static void process(int samples[], int sampleRate)
 26     {
 27        // Here, we make the sound three times as loud
 28        // In this example, we don't need the sample rate
 29        
 30        for (int i = 0; i < samples.length; i++)
 31        {
 32           samples[i] = 3 * samples[i];
 33        }
 34     }

 38     /*
 39       The code below processes a file in the WAV format.
 40       You can use this program to manipulate sound files without 
 41       reading or understanding the code below. 
 42       
 43       The code uses Java features that are introduced in chapter 19, 
 44       as well as the internals of the WAV format 
 45       (https://ccrma.stanford.edu/courses/422/projects/WaveFormat/)
 46     */

 48     /**
 49        Gets an unsigned 4-byte integer from a random access file.
 50        @param in the file
 51        @return the integer 
 52     */
 53     public static int getUnsignedInt4(RandomAccessFile in) 
 54        throws IOException
 55     {
 56        int result = 0;
 57        int base = 1;
 58        for (int i = 0; i < 4; i++)
 59        {
 60           result = result + in.read() * base;
 61           base = base * 256;
 62        }
 63        return result;
 64     }
 65  
 66     /**
 67        Gets an unsigned 2-byte integer from a random access file.
 68        @param in the file
 69        @return the integer
 70     */
 71     public static int getUnsignedInt2(RandomAccessFile in)
 72        throws IOException
 73     {
 74        int lo = in.read();
 75        int hi = in.read();
 76        return lo + 256 * hi;
 77     }
 78  
 79     /**
 80        Gets a signed 2-byte integer from a random access file.
 81        @param in the file
 82        @return the integer
 83     */
 84     public static int getSignedInt2(RandomAccessFile in)
 85        throws IOException
 86     {
 87        int lo = in.read();
 88        int hi = in.read();
 89        int result = lo + 256 * hi;
 90        if (result >= 32768) { result = result - 65536; }
 91        return result;
 92     }
 93  
 94     /**
 95        Puts a signed 2-byte integer to a random access file.
 96        @param out the file
 97        @param value the integer to put
 98     */
 99     public static void putSignedInt2(RandomAccessFile out, int value)
100        throws IOException
101     {
102        if (value < 0) { value = value + 65536; }
103        out.write(value % 256);
104        out.write(value / 256);
105     }

107     public static void main(String[] args) throws Exception
108     {
109        JFileChooser chooser = new JFileChooser(".");
110        if (chooser.showOpenDialog(null) != JFileChooser.APPROVE_OPTION) 
111        {
112           System.out.println("No sound file selected.");
113           System.exit(1);
114        }
115  
116        RandomAccessFile file = new RandomAccessFile(
117           chooser.getSelectedFile(), "rw");
118  
119        // Check that we can handle this file
120        file.seek(20);
121        int formatType = getUnsignedInt2(file);
122        if (formatType != 1)
123        {
124           System.out.println("Not an uncompressed sound file.");
125           System.exit(1);
126        }
127        int numChannels = getUnsignedInt2(file);
128        if (numChannels != 1)
129        {
130           System.out.println("Not a mono sound file.");
131           System.exit(1);
132        }
133  
134        final int sampleRate = getUnsignedInt2(file);
135  
136        file.seek(34);
137        int bitsPerSample = getUnsignedInt2(file);
138        if (bitsPerSample != 16)
139        {
140           System.out.println("Not a 16 bit sound file.");
141           System.exit(1);
142        }
143  
144        // Read data size and allocate data array
145        file.seek(40);
146        int dataSize = getUnsignedInt4(file) / 2; // 2 bytes per data
147        final int[] samples = new int[dataSize];
148  
149        // Read sound data
150        for (int i = 0; i < dataSize; i++) 
151        {
152           samples[i] = getSignedInt2(file);
153        }
154  
155        final int[] original = Arrays.copyOf(samples, dataSize);
156  
157        // Process sound data
158        process(samples, sampleRate);
159  
160        // Write sound data
161        file.seek(44);
162        for (int i = 0; i < dataSize; i++)
163        {
164           putSignedInt2(file, samples[i]);
165        }
166  
167        file.close();
168  
169        // This is the end of the file processing. 
170  
171        // Next, we show the original and processed wave forms
172  
173        JFrame frame = new JFrame();
174  
175        final int FRAME_WIDTH = 800;
176        final int FRAME_HEIGHT = 200;
177  
178        frame.setSize(FRAME_WIDTH, FRAME_HEIGHT);
179        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
180  
181        JComponent component = new JComponent()
182        {
183            public void paintComponent(Graphics graph)
184            {
185               int increment = samples.length / getWidth();
186  
187               // Find the largest value for scaling the displayed values
188               int largest = 0;
189               for (int i = 0; i < samples.length; i = i + increment)
190               {
191                  int sample = Math.abs(samples[i]);
192                  int orig = Math.abs(original[i]);
193                  if (sample > largest) { largest = sample; } 
194                  if (orig > largest) { largest = orig; }
195               }
196  
197               int x = 0;
198               for (int i = 0; i < samples.length; i = i + increment)
199               {
200                  int height = getHeight() / 4;
201                  int y = height - original[i] * height / largest;
202                  graph.drawLine(x, y, x, height);
203                  y = 3 * height - samples[i] * height / largest;
204                  graph.drawLine(x, y, x, 3 * height);
205                  x++;
206               }
207            }
208        };
209        
210        frame.add(component);
211        frame.setVisible(true);
212  
213        // Now play the sound
214  
215        Clip clip = AudioSystem.getClip();
216        clip.open(AudioSystem.getAudioInputStream(
217              chooser.getSelectedFile()));
218        clip.start(); 
219     }
