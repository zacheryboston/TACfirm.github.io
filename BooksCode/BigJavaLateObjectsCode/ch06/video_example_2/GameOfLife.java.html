  5     /**
  6        Counts the live neighbors of a given location.
  7        @param cells the array of cells
  8        @param i the row index of the location
  9        @param j the column index of the location
 10        @return the number of live neighbors
 11     */
 12     public static int countNeighbors(boolean[][] cells, int i, int j)
 13     {
 14        int count = 0;
 15        for (int di = -1; di <= 1; di++)
 16        {
 17           for (int dj = -1; dj <= 1; dj++)
 18           {
 19              if (di != 0 || dj != 0) // Don't count the given location
 20              {
 21                 int row = i + di;
 22                 int column = j + dj;
 23                 if (0 <= row && row < cells.length
 24                       // row index valid
 25                    && 0 <= column && column < cells[0].length
 26                       // column index valud
 27                    && cells[row][column]) // cell is live
 28                 {
 29                    count++;
 30                 }
 31              }
 32           }
 33        }
 34        return count;
 35     }

 37     public static boolean[][] nextGeneration(boolean[][] current)
 38     {
 39        int rows = current.length;
 40        int columns = current[0].length;
 41  
 42        boolean[][] next = new boolean[rows][columns];
 43  
 44        for (int i = 0; i < rows; i++)
 45        {
 46           for (int j = 0; j < columns; j++)
 47           {
 48              int count = countNeighbors(current, i, j);
 49              if (!current[i][j] && count == 3) 
 50                 // Birth
 51              {
 52                 next[i][j] = true;
 53              }
 54              else if (current[i][j] && count >= 2 && count <= 3) 
 55                 // Survival
 56              {
 57                 next[i][j] = true;
 58              }
 59           }
 60        }      
 61        return next;
 62     }

 68     public static void init(boolean[][] cells)
 69     {
 70        // Place the initial configuration in the middle
 71  
 72        int i = cells.length / 2;
 73        int j = cells[0].length / 2;
 74  
 75        cells[i - 1][j-1] = true;
 76        cells[i - 1][j] = true;
 77        cells[i - 1][j + 1] = true;
 78        cells[i][j + 1] = true;
 79        cells[i + 1][j] = true;
