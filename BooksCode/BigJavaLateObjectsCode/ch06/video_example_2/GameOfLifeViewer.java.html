1  import java.awt.Dimension;
  2  import java.awt.Graphics;
  3  import java.awt.Rectangle;
  4  import javax.swing.JComponent;
  5  import javax.swing.JFrame;
  6  
  7  public class GameOfLifeViewer extends JFrame
  8  {
  9     private boolean[][] cells;
 10  
 11     public GameOfLifeViewer(final int rows, final int columns)
 12     {
 13        final int CELL_SIZE = 8;      
 14        final Dimension preferredSize = new Dimension(
 15           columns * CELL_SIZE,
 16           rows * CELL_SIZE);
 17  
 18        add(new JComponent() 
 19           {
 20              public void paintComponent(Graphics g)
 21              {
 22                 for (int i = 0; i < cells.length; i++)
 23                 {
 24                    for (int j = 0; j < cells[0].length; j++)
 25                    {
 26                       if (cells[i][j])
 27                       {
 28                          g.fillRect(j * CELL_SIZE, i * CELL_SIZE,
 29                             CELL_SIZE - 1, CELL_SIZE - 1);
 30                       }
 31                    }                  
 32                 }
 33              }
 34              
 35              public Dimension getPreferredSize()
 36              {
 37                 return preferredSize;
 38              }
 39           }); 
 40        pack();
 41        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
 42        setVisible(true);
 43     }
 44  
 45     public void update(boolean[][] newCells)
 46     {
 47        cells = newCells;
 48        repaint();
 49        try
 50        {
 51           Thread.sleep(100);         
 52        }
 53        catch (InterruptedException ex)
 54        {
 55        }
 56     }
 57  }
