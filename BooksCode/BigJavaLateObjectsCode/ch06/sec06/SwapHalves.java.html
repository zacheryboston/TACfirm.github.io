1  import java.util.Arrays;
  2  
  3  public class SwapHalves
  4  {
  5     /**
  6        Swaps the elements of an array at given positions.
  7        @param a the array 
  8        @param i the first position
  9        @param j the second position
 10     */
 11     public static void swap(int[] a, int i, int j)
 12     {
 13        int temp = a[i];
 14        a[i] = a[j];
 15        a[j] = temp;
 16     }
 17  
 18     public static void main(String[] args)
 19     {
 20        int[] values = { 9, 13, 21, 4, 11, 7, 1, 3 };
 21  
 22        int i = 0;
 23        int j = values.length / 2;
 24        while (i < values.length / 2)
 25        {
 26           swap(values, i, j);
 27           i++;
 28           j++;
 29        }
 30  
 31        System.out.println(Arrays.toString(values));
 32     }
 33  }
