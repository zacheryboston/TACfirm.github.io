  4       * Removes duplicates from an array.
  5       * @param a an array

  7       * @return the new size
  8       */
  9      public static int removeDuplicates(String[] a, int currentSize)
 10      {
 11          int start = 0;
 12          while (start < currentSize)
 13          {
 14              int removed = removeFromTail(a, currentSize, 
 15                      start + 1, a[start]);
 16              currentSize = currentSize - removed;
 17              start++;
 18          }
 19          return currentSize;
 20      }

 27       * @param toBeRemoved the value to be removed
 28       * @return the number of elements removed
 29       */
 30      public static int removeFromTail(String[] a, int currentSize, 
 31      int start, String toBeRemoved)
 32      {
 33          int source = start; 
 34          int target = start;
 35          int removed = 0;
 36          while (source < currentSize)
 37          {
 38              if (a[source].equals(toBeRemoved))
 39              {
 40                  removed++;
 41              }
 42              else
 43              {
 44                  a[target] = a[source];
 45                  target++;
 46              }
 47              source++;
 48          }
 49          return removed;
 50      }
