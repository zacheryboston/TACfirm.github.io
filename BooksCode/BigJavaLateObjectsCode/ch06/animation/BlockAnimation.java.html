1  import java.awt.Color;
  2  import java.awt.Graphics;
  3  import java.awt.event.ActionEvent;
  4  import java.awt.event.ActionListener;
  5  import javax.swing.JFrame;
  6  import javax.swing.JComponent;
  7  import javax.swing.Timer;
  8  
  9  /*
 10     A drawing of a moving block.
 11  */
 12  public class BlockAnimation
 13  {  
 14     public static void draw(Graphics g, int frame)
 15     {  
 16        final int BLOCK_WIDTH = 20;
 17        final int BLOCK_HEIGHT = 20;
 18        g.fillRect(frame, 0, BLOCK_WIDTH, BLOCK_HEIGHT);
 19     }
 20  
 21     public static void main(String[] args)
 22     {
 23        // Do not look at the code in the main method
 24        // Your code will go into the draw method above
 25         
 26        JFrame frame = new JFrame();
 27  
 28        final int FRAME_WIDTH = 400;
 29        final int FRAME_HEIGHT = 400;
 30  
 31        frame.setSize(FRAME_WIDTH, FRAME_HEIGHT);
 32        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
 33  
 34        final JComponent component = new JComponent()
 35        {
 36           private int frame = 0;
 37  
 38           public void paintComponent(Graphics graph)
 39           {
 40              frame++;
 41              draw(graph, frame);
 42           }
 43        };
 44  
 45        ActionListener timerListener = new ActionListener()
 46        {
 47           public void actionPerformed(ActionEvent event)
 48           {
 49              component.repaint();
 50           }
 51        };
 52  
 53        final int DELAY = 25; // 25 milliseconds between frames
 54        Timer t = new Timer(DELAY, timerListener);
 55        t.start();
 56        
 57        frame.add(component);
 58        frame.setVisible(true);
 59     }   
 60  }
