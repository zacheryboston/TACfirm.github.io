1  import java.util.List;
  2  import java.util.stream.Collectors;
  3  import java.util.stream.Stream;
  4  
  5  public class Infinite
  6  {
  7     public static void main(String[] args)
  8     {
  9        Stream<Integer> ones = Stream.generate(() -> 1);
 10        show("An infinite stream of ones", ones);
 11  
 12        Stream<Integer> dieTosses =
 13           Stream.generate(() -> 1 + (int)(6 * Math.random()));
 14        show("An infinite stream of random numbers", dieTosses);
 15  
 16        Stream<Integer> integers = Stream.iterate(0, n -> n + 1);
 17        show("An infinite stream of integers", integers);
 18  
 19        integers = Stream.iterate(0, n -> n + 1);
 20        Stream<Integer> primes = integers.filter(n -> isPrime(n));
 21        show("An infinite stream of primes", primes);      
 22     }
 23     
 24     public static boolean isPrime(int number)
 25     {
 26        return Stream.iterate(1, n -> n + 1)
 27           .limit(number)
 28           .filter(n -> number % n == 0)
 29           .count() == 2;
 30     }
 31  
 32     /**
 33        Displays the first ten values in a stream, followed by ...
 34        if there are additional values.
 35        @param caption the caption preceding the values
 36        @param stream the stream to be displayed (which is consumed
 37        in the process)
 38     */
 39     public static <T> void show(String caption, Stream<T> stream)
 40     {
 41        final int SHOW_LIMIT = 10;
 42        List<T> values = stream.limit(SHOW_LIMIT + 1).collect(
 43           Collectors.toList());
 44        System.out.print(caption + ": ");
 45        for (int i = 0; i < Math.min(SHOW_LIMIT, values.size()); i++)
 46        {
 47           if (i > 0) { System.out.print(", "); }
 48           System.out.print(values.get(i));
 49        }
 50        if (values.size() > SHOW_LIMIT) { System.out.print(", ..."); }
 51        System.out.println();
 52     }
 53  }
