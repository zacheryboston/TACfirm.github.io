1  import java.util.Optional;
  2  import java.util.stream.Stream;
  3  
  4  public class OptionalDemo
  5  {
  6     public static void main(String[] args)
  7     {
  8        Stream<String> words = Stream.of("Zimbabwe", "Namibia",
  9           "Uganda", "Botswana", "Madagascar", "Mozambique",
 10           // "Ivory Coast",
 11           "Swaziland", "Togo");
 12        Optional<String> optResult = words.filter(w -> w.length() > 10).findFirst();
 13  
 14        // Using orElse
 15        String first = optResult.orElse("(None)");
 16        System.out.println("The first country with more than ten characters: " + first);
 17  
 18        // Using ifPresent
 19        optResult.ifPresent(v -> System.out.println(
 20              "The first country with more than ten characters: " + v));
 21  
 22        // Using isPresent/get
 23        if (optResult.isPresent())
 24        {
 25           System.out.println("The first country with more than ten characters: " + optResult.get());
 26        }
 27        else
 28        {
 29           System.out.println("None of these countries has more than ten characters");
 30        }
 31     }
 32  }
