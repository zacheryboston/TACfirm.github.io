1  import java.util.Arrays;
  2  import java.util.Random;
  3  import java.util.stream.DoubleStream;
  4  import java.util.stream.IntStream;
  5  import java.util.stream.Stream;
  6  
  7  public class PrimitiveStreams
  8  {
  9     public static void main(String[] args)
 10     {
 11        IntStream stream = IntStream.of(3, 1, 4, 1, 5, 9);
 12        System.out.println("The sum of a stream of ints: " + stream.sum());
 13  
 14        int[] values = { 3, 1, 4, 1, 5, 9 };
 15        stream = IntStream.of(values);
 16        System.out.println("The sum of a stream of ints from an array: "
 17           + stream.sum());
 18  
 19        stream = IntStream.range(1, 101);
 20        System.out.println("The sum of a range: " + stream.sum());
 21  
 22        Random generator = new Random();
 23        IntStream dieTosses = generator.ints(1, 7);
 24        System.out.println("Twenty random die tosses: "
 25           + Arrays.toString(dieTosses.limit(20).toArray()));
 26  
 27        IntStream codePoints = "Hello".codePoints();
 28        System.out.println("The Unicode code points of the string Hello: "
 29           + Arrays.toString(codePoints.toArray()));
 30  
 31        Stream<String> words = Stream.of("Hello", "World");
 32        stream = words.mapToInt(w -> w.length());
 33        System.out.println("Mapping objects to ints: " 
 34           + Arrays.toString(stream.toArray()));
 35  
 36        stream = IntStream.range(0, 20).map(n -> Math.max(n, 10));
 37        System.out.println("Mapping ints to ints: "
 38           + Arrays.toString(stream.toArray()));
 39  
 40        String river = "Mississippi";
 41        int n = river.length();
 42        Stream<String> prefixes
 43           = IntStream.range(0, n).mapToObj(i -> river.substring(0, i + 1));
 44        System.out.println("Mapping ints to objects: " 
 45           + Arrays.toString(prefixes.toArray(String[]::new)));
 46        
 47        DoubleStream roots = IntStream.range(0, 20).mapToDouble(i -> Math.sqrt(i));
 48        System.out.println("Mapping ints to doubles: "
 49           + Arrays.toString(roots.toArray()));
 50  
 51        words = Stream.of("Goodbye", "World");
 52        double average = words
 53           .mapToInt(w -> w.length())
 54           .average()
 55           .orElse(0);
 56        System.out.println("Computing an average of ints: " + average);
 57  
 58        int max = "World".codePoints()
 59           .max()
 60           .orElse(0);
 61  
 62        System.out.println("Computing a maximum of ints: " + max);
 63     }
 64  }
