1  import java.util.List;
  2  import java.util.stream.Collectors;
  3  import java.util.stream.Stream;
  4  
  5  public class Streams
  6  {
  7     public static void main(String[] args)
  8     {
  9        Stream<String> words = Stream.of("A", "Tale", "of", "Two", "Cities");
 10        Stream<String> lowerCaseWords = words.map(w -> w.toLowerCase());
 11        show("Mapping to lowercase: ", lowerCaseWords);
 12  
 13        lowerCaseWords = Stream.of("A", "Tale", "of", "Two", "Cities")
 14           .map(w -> w.toLowerCase());
 15  
 16        Stream<String> consonantsOnly = lowerCaseWords.map(w -> w.replaceAll("[aeiou]", ""));
 17        show("Removing vowels", consonantsOnly);
 18  
 19        consonantsOnly = Stream.of("A", "Tale", "of", "Two", "Cities")
 20           .map(w -> w.toLowerCase())
 21           .map(w -> w.replaceAll("[aeiou]", ""));
 22  
 23        Stream<Integer> consonantCount = consonantsOnly.map(w -> w.length());
 24        show("Counting consonants", consonantCount);
 25  
 26        words = Stream.of(
 27           "how much wood could a wood chuck chuck".split(" "));
 28        Stream<String> cWords = words.filter(w -> w.substring(0, 1).equals("c"));
 29        show("Words with c", cWords);
 30  
 31        words = Stream.of(
 32           "how much wood could a wood chuck chuck".split(" "));
 33        Stream<String> distinctWords = words.distinct();
 34        show("Distinct words", distinctWords);
 35  
 36        distinctWords = Stream.of(
 37           "how much wood could a wood chuck chuck".split(" "))
 38           .distinct();
 39        Stream<String> sortedWords = distinctWords.sorted();
 40        show("Sorted words", sortedWords);
 41     }
 42     
 43     /**
 44        Displays the first ten values in a stream, followed by ...
 45        if there are additional values.
 46        @param caption the caption preceding the values
 47        @param stream the stream to be displayed (which is consumed
 48        in the process)
 49     */
 50     public static <T> void show(String caption, Stream<T> stream)
 51     {
 52        final int SHOW_LIMIT = 10;
 53        List<T> values = stream.limit(SHOW_LIMIT + 1).collect(
 54           Collectors.toList());
 55        System.out.print(caption + ": ");
 56        for (int i = 0; i < Math.min(SHOW_LIMIT, values.size()); i++)
 57        {
 58           if (i > 0) { System.out.print(", "); }
 59           System.out.print(values.get(i));
 60        }
 61        if (values.size() > SHOW_LIMIT) { System.out.print(", ..."); }
 62        System.out.println();
 63     }
 64  }
