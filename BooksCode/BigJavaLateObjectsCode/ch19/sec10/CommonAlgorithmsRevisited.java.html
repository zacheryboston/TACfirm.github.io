1  import java.util.Arrays;
  2  import java.util.OptionalDouble;
  3  import java.util.stream.IntStream;
  4  import java.util.stream.DoubleStream;
  5  import java.util.stream.Collectors;
  6  
  7  public class CommonAlgorithmsRevisited
  8  {
  9     public static void main(String[] args)
 10     {
 11        int n = 10;
 12        int[] squares = IntStream.range(0, n)
 13           .map(i -> i * i)
 14           .toArray();
 15  
 16        System.out.println(Arrays.toString(squares));
 17  
 18        double[] values = { 32, 54, 67.5, 29, 35, 80, 115, 44.5, 100, 65 };
 19        double total = DoubleStream.of(values).sum();
 20        double average = DoubleStream.of(values).average().orElse(0);
 21        double largest = DoubleStream.of(values).max().orElse(Double.MIN_VALUE);
 22        double smallest = DoubleStream.of(values).min().orElse(Double.MAX_VALUE);
 23        System.out.println("total: " + total);
 24        System.out.println("average: " + average);
 25        System.out.println("largest: " + largest);
 26        System.out.println("smallest: " + smallest);
 27  
 28        String str = "How much wood could a woodchuck chuck?";
 29        long spaces = str.codePoints()
 30           .filter(ch -> ch == ' ')
 31           .count();
 32        System.out.println("spaces: " + spaces);
 33  
 34        String result = DoubleStream.of(values)
 35           .mapToObj(v -> "" + v)
 36           .collect(Collectors.joining(" | "));
 37        System.out.println(result);
 38  
 39        OptionalDouble result2 = DoubleStream.of(values)
 40           .filter(v -> v > 100)
 41           .findFirst();
 42        System.out.println(result2);
 43  
 44        n = values.length;
 45        int pos = IntStream.range(0, n)
 46           .filter(i -> values[i] == 100)
 47           .findFirst()
 48           .orElse(-1);
 49        System.out.println("pos: " + pos);
 50     }
 51  }
