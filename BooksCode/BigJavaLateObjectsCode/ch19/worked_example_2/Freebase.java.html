 23     public static void main(String[] args) throws Exception
 24     {
 25        String apiKey = args[0];
 26        String urlString = "https://www.googleapis.com/freebase/v1/mqlread";
 27        String query = "[{"
 28           + "  \"type\": \"/film/film\","
 29           + "  \"name\": null,"
 30           + "  \"initial_release_date\": null,"
 31           + "  \"directed_by\": [],"
 32           + "  \"produced_by\": [],"
 33           + "  \"starring\": [{"
 34           + "    \"actor\": []"
 35           + "  }],"
 36           + "  \"country\": [{"
 37           + "    \"id\": \"/en/united_states\""
 38           + "  }],"
 39           + "  \"genre\": [{"
 40           + "    \"id\": \"/en/drama\""
 41           + "  }]"
 42           + "}]";
 43        String cursor = null;
 44        boolean done = false;
 45        while (!done)
 46        {
 47           String request = "?query=" + URLEncoder.encode(query, "UTF-8") + "&key=" + apiKey + "&cursor" + (cursor == null ? "" : "=" + URLEncoder.encode(cursor, "UTF-8"));
 48           URL url = new URL(urlString + request);
 49           try (InputStream in = url.openStream()) 
 50           {
 51              JsonReader reader = Json.createReader(in);
 52              JsonObject contents = reader.readObject();
 53              JsonArray result = contents.getJsonArray("result");
 54              for (int k = 0; k < result.size(); k++)
 55              {
 56                 printFilm(result.getJsonObject(k));
 57              }
 58              done = result.size() < 100;
 59              if (!done) { cursor = contents.getString("cursor"); }
 60           }
 61        }
 62     }

 64     public static StringBuilder format(String title, List<JsonValue> array) 
 65     {
 66        StringBuilder result = new StringBuilder();
 67        result.append(title);
 68        result.append(": ");
 69        for (int i = 0; i < array.size(); i++) 
 70        {
 71           if (i > 0) { result.append(", "); }
 72           result.append(array.get(i).toString().replace(",", " "));
 73        }
 74        return result;
 75     }
 76  
 77     public static void printFilm(JsonObject film)
 78     {
 79        System.out.println("Name: " + film.getString("name"));
 80        System.out.println("Year: " + film.get("initial_release_date").toString().replace("\"", "").substring(0, 4));      
 81        System.out.println(format("Directed by", film.getJsonArray("directed_by")));
 82        System.out.println(format("Produced by", film.getJsonArray("produced_by")));
 83        List<JsonValue> actors = new ArrayList<>();
 84        JsonArray jactors = film.getJsonArray("starring");
 85        for (int i = 0; i < jactors.size(); i++)
 86        {
 87           actors.addAll(jactors.getJsonObject(i).getJsonArray("actor"));
 88        }
 89        System.out.println(format("Actors", actors));
 90     }
