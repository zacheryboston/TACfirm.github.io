 18     public static List<Movie> readMovies(String filename) throws IOException
 19     {
 20        List<Movie> movies = new ArrayList<>();
 21        try (Scanner in = new Scanner(new File(filename))) 
 22        {
 23           while (in.hasNextLine()) 
 24           {
 25              String nameLine = in.nextLine();
 26              String yearLine = in.nextLine();
 27              String directorsLine = in.nextLine();
 28              String producersLine = in.nextLine();
 29              String actorsLine = in.nextLine();
 30              movies.add(new Movie(getString(nameLine), 
 31                    Integer.parseInt(getString(yearLine)),
 32                    getList(directorsLine), getList(producersLine),
 33                    getList(actorsLine)));
 34           }
 35        }
 36        return movies;
 37     }
 38  
 39     private static String getString(String line)
 40     {
 41        int colon = line.indexOf(":");
 42        return line.substring(colon + 1).trim();
 43     }
 44  
 45     private static List<String> getList(String line)
 46     {
 47        return Stream.of(getString(line).split(", "))
 48           .collect(Collectors.toList());
 49        /* Alternative without streams:
 50        ArrayList<String> result = new ArrayList<>();
 51        for (String s : getString(line).split(", "))
 52        {
 53           result.add(s);
 54        }
 55        return result;
 56        */
 57     }

 60     {
 61        List<Movie> movieList = readMovies("movies.txt");

 65           .collect(Collectors.toList());
 66        System.out.println("Movies that start with X: "

 79           + count);

 82           .mapToInt(m -> m.getActors().size())

126     public static boolean commonActorAndDirector(Movie m)
127     {
128        return m.getDirectors().stream().anyMatch(d -> m.getActors().contains(d));
129     }
130  	
131     public static Set<String> intersect(Collection<String> a, Collection<String> b) 
132     {
133        Set<String> intersection = new HashSet<>(a);
134        intersection.retainAll(b);
135        return intersection;
136     }
137     
138     public static boolean isSorted(List<String> names) 
139     {
140        List<String> lastNames = names.stream().map(n -> n.substring(n.lastIndexOf(" ") + 1)).collect(Collectors.toList());
141        return lastNames.equals(lastNames.stream().sorted().collect(Collectors.toList()));
142     }
