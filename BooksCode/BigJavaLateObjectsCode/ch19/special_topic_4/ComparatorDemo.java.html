1  import java.io.IOException;
  2  import java.nio.file.Files;
  3  import java.nio.file.Paths;
  4  import java.util.Arrays;
  5  import java.util.Comparator;
  6  import java.util.List;
  7  import java.util.stream.Collectors;
  8  import java.util.stream.Stream;
  9  
 10  
 11  public class ComparatorDemo
 12  {
 13     public static void main(String[] args) throws IOException
 14     {
 15        Comparator<String> comp = Comparator.comparing(t -> t.length());
 16        String[] words = "how much wood could a wood chuck chuck".split(" ");
 17        Arrays.sort(words, comp);
 18        System.out.println(Arrays.toString(words));
 19  
 20        Arrays.sort(words, Comparator.comparing(String::length));
 21        System.out.println(Arrays.toString(words));
 22  
 23        try (Stream<String> lines = Files.lines(Paths.get("../population.txt")))
 24        {
 25           List<Country> result = lines
 26              .map(line -> Country.parse(line))
 27              .sorted(Comparator
 28                 .comparing(Country::getContinent)
 29                 .thenComparing(Country::getName))
 30              .collect(Collectors.toList());
 31           System.out.println(result);
 32        }
 33     }
 34  }
