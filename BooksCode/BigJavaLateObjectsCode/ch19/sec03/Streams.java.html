1  import java.util.Arrays;
  2  import java.util.List;
  3  import java.util.Set;
  4  import java.util.stream.Collectors;
  5  import java.util.stream.Stream;
  6  
  7  public class Streams
  8  {
  9     public static void main(String[] args)
 10     {
 11        Stream<String> stream = Stream
 12           .of("Mary", "had", "a", "little", "lamb")
 13           .filter(w -> w.contains("a"));
 14  
 15        String[] result1 = stream.toArray(String[]::new);
 16        System.out.println("Converting to an array: "
 17           + Arrays.toString(result1));
 18        
 19        stream = Stream
 20           .of("Mary", "had", "a", "little", "lamb")
 21           .filter(w -> w.contains("a"));
 22        
 23        List<String> result2 = stream.collect(Collectors.toList());
 24        System.out.println("Collecting to a list: " + result2);
 25        
 26        stream = Stream
 27           .of("Mary", "had", "a", "little", "lamb")
 28           .filter(w -> w.contains("a"));
 29  
 30        Set<String> result3 = stream.collect(Collectors.toSet());
 31        System.out.println("Collecting to a set: " + result3);
 32        
 33        stream = Stream
 34           .of("Mary", "had", "a", "little", "lamb")
 35           .filter(w -> w.contains("a"));
 36        String result4 = stream.collect(Collectors.joining(", "));
 37        System.out.println("Joining with commas: " + result4);
 38     }
 39  }
