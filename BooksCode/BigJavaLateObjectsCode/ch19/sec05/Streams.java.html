1  import java.util.List;
  2  import java.util.stream.Collectors;
  3  import java.util.stream.Stream;
  4  
  5  public class Streams
  6  {
  7     public static void main(String[] args)
  8     {
  9        Stream<String> words = Stream.of(
 10           "how much wood could a wood chuck chuck".split(" "))
 11           .filter(w -> w.length() > 4);
 12        show("Filtering with a simple lambda expression", words);
 13  
 14        words = Stream.of(
 15           "how much wood could a wood chuck chuck".split(" "))
 16           .filter((String w) -> w.length() > 4);
 17        show("Filtering with a lambda expression with parameter type", words);
 18  
 19        Stream<String> distinctWords = Stream.of(
 20           "how much wood could a wood chuck chuck".split(" "))
 21           .distinct();
 22        Stream<String> sortedWords = distinctWords.sorted(
 23           (s, t) -> s.length() - t.length());
 24        show("Sorting with a lambda expression with two parameters", sortedWords);
 25        
 26        distinctWords = Stream.of(
 27           "how much wood could a wood chuck chuck".split(" "))
 28           .distinct();
 29        sortedWords = distinctWords.sorted(
 30           (v, w) ->
 31           {
 32              int first = v.length();
 33              int second = w.length();
 34              return first - second;
 35           });
 36        show("Sorting with a complex lambda expression", sortedWords);      
 37     }
 38  
 39     /**
 40        Displays the first ten values in a stream, followed by ...
 41        if there are additional values.
 42        @param caption the caption preceding the values
 43        @param stream the stream to be displayed (which is consumed
 44        in the process)
 45     */
 46     public static <T> void show(String caption, Stream<T> stream)
 47     {
 48        final int SHOW_LIMIT = 10;
 49        List<T> values = stream.limit(SHOW_LIMIT + 1).collect(
 50           Collectors.toList());
 51        System.out.print(caption + ": ");
 52        for (int i = 0; i < Math.min(SHOW_LIMIT, values.size()); i++)
 53        {
 54           if (i > 0) { System.out.print(", "); }
 55           System.out.print(values.get(i));
 56        }
 57        if (values.size() > SHOW_LIMIT) { System.out.print(", ..."); }
 58        System.out.println();
 59     }
 60  }
