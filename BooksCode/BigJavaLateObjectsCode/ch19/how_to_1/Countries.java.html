1  import java.io.IOException;
  2  import java.nio.file.Files;
  3  import java.nio.file.Paths;
  4  import java.util.List;
  5  import java.util.Map;
  6  import java.util.Optional;
  7  import java.util.stream.Stream;
  8  import java.util.stream.Collectors;
  9  
 10  public class Countries
 11  {
 12     public static void main(String[] args) throws IOException
 13     {
 14        try (Stream<String> lines = Files.lines(Paths.get("../population.txt"))) {
 15           Stream<Country> countries = 
 16              lines.map(line -> Country.parse(line));
 17           List<Country> countryList = countries.collect(Collectors.toList());
 18  
 19           Stream<Country> africanCountries = countryList.stream()
 20              .filter(c -> c.getContinent().equals("Africa"));
 21  
 22           System.out.println("African countries: "
 23              + africanCountries.collect(Collectors.toList()));
 24        
 25           Stream<String> continents = countryList.stream()
 26              .map(c -> c.getContinent())
 27              .distinct();
 28        
 29           System.out.println("Continents: "
 30              + continents.collect(Collectors.toList()));
 31        
 32           Stream<String> continentsWithPopulousCountries = countryList.stream()
 33              .filter(c -> c.getPopulation() >= 100_000_000)
 34              .map(c -> c.getContinent())
 35              .distinct();
 36  
 37           System.out.println("Continents with populous countries: "
 38              + continentsWithPopulousCountries.collect(Collectors.toList()));
 39  
 40           Optional<Country> aPopulousCountry = countryList.stream()
 41              .filter(country -> country.getPopulation() > 100_000_000)
 42              .findAny();
 43        
 44           aPopulousCountry.ifPresent(
 45              c -> System.out.println("A populous country: " + c));
 46        
 47           List<Country> mostPopulous = countryList.stream()
 48              .sorted((c, d) -> Double.compare(d.getPopulation(), c.getPopulation()))
 49              .limit(10)
 50              .collect(Collectors.toList());
 51  
 52           System.out.println("Ten most populous countries: "
 53              + mostPopulous);
 54  
 55           double average = countryList.stream()
 56              .mapToInt(country -> country.getPopulation())
 57              .average()
 58              .orElse(0);
 59  
 60           System.out.println("Average population of all countries: "
 61              + average);
 62  
 63           Map<String, List<Country>> countriesByContinent = countryList.stream()
 64              .collect(
 65                 Collectors.groupingBy(c -> c.getContinent()));
 66  
 67           System.out.println("Countries by continent: "
 68              + countriesByContinent);
 69  
 70           Map<String, Double> averagePopulationByContinent = countryList.stream()
 71              .collect(
 72                 Collectors.groupingBy(
 73                    c -> c.getContinent(),
 74                    Collectors.averagingInt(c -> c.getPopulation())));
 75  
 76           System.out.println("Average population by continent: "
 77              + averagePopulationByContinent);
 78        }
 79     }
 80  }
