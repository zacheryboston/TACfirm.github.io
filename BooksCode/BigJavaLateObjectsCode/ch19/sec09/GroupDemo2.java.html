1  import java.io.IOException;
  2  import java.nio.file.Files;
  3  import java.nio.file.Paths;
  4  import java.util.Map;
  5  import java.util.stream.Stream;
  6  import java.util.stream.Collectors;
  7  
  8  public class GroupDemo2
  9  {
 10     public static void main(String[] args) throws IOException
 11     {
 12        try (Stream<String> lines = Files.lines(Paths.get("../population.txt")))
 13        {
 14           Stream<Country> countries
 15              = lines.map(line -> Country.parse(line));
 16  
 17           Map<String, Long> groupSum = countries.collect(
 18              Collectors.groupingBy(
 19                 c -> c.getContinent(), // The function for extracting the keys
 20                 Collectors.summingLong(
 21                    c -> c.getPopulation()))); // The function for getting the summands
 22  
 23           System.out.println("Population per continent: "
 24              + groupSum);
 25        }
 26     }
 27  }
