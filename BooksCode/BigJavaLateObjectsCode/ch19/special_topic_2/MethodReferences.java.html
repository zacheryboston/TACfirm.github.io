1  import java.util.Arrays;
  2  import java.util.List;
  3  import java.util.stream.Collectors;
  4  import java.util.stream.Stream;
  5  
  6  public class MethodReferences
  7  {
  8     public static void main(String[] args)
  9     {
 10        Stream<String> words = Stream.of(
 11           "how much wood could a wood chuck chuck".split(" "))
 12           .map(w -> w.toUpperCase());
 13        show("Mapping to uppercase with a lambda expression", words);
 14  
 15        words = Stream.of(
 16           "how much wood could a wood chuck chuck".split(" "))
 17           .map(String::toUpperCase);
 18        show("Mapping to uppercase with a method reference", words);
 19  
 20        words = Stream.of(
 21           "how much wood could a wood chuck chuck".split(" "))
 22           .sorted(String::compareTo);
 23        show("Sorting strings with a method reference", words);
 24  
 25        Stream<Double> numbers = Stream.of(32.0, 54.0, 67.5, 29.0, 35.0, 80.0, 115.0, 44.5, 100.0, 65.0)
 26           .sorted(Double::compare);
 27        show("Sorting numbers with a method reference", numbers);
 28  
 29        Stream<BankAccount> accounts = Stream
 30           .generate(BankAccount::new)
 31           .limit(5);
 32        show("Constructing accounts with a constructor reference", accounts);
 33        
 34        accounts = Stream
 35           .iterate(0.0, n -> n + 1000.0)
 36           .map(BankAccount::new)
 37           .limit(5);
 38        show("Mapping with a constructor reference", accounts);
 39  
 40        words = Stream.of(
 41           "how much wood could a wood chuck chuck".split(" "));
 42        String[] array = words.toArray(String[]::new);
 43        System.out.println(
 44           "Using toArray with an array constructor reference: "
 45           + Arrays.toString(array));
 46     }
 47     
 48     /**
 49        Displays the first ten values in a stream, followed by ...
 50        if there are additional values.
 51        @param caption the caption preceding the values
 52        @param stream the stream to be displayed (which is consumed
 53        in the process)
 54     */
 55     public static <T> void show(String caption, Stream<T> stream)
 56     {
 57        final int SHOW_LIMIT = 10;
 58        List<T> values = stream.limit(SHOW_LIMIT + 1).collect(
 59           Collectors.toList());
 60        System.out.print(caption + ": ");
 61        for (int i = 0; i < Math.min(SHOW_LIMIT, values.size()); i++)
 62        {
 63           if (i > 0) { System.out.print(", "); }
 64           System.out.print(values.get(i));
 65        }
 66        if (values.size() > SHOW_LIMIT) { System.out.print(", ..."); }
 67        System.out.println();
 68     }
 69  }
