1  import java.io.IOException;
  2  import java.nio.file.Files;
  3  import java.nio.file.Paths;
  4  import java.util.Comparator;
  5  import java.util.List;
  6  import java.util.stream.Stream;
  7  import java.util.stream.Collectors;
  8  
  9  public class Words
 10  {      
 11     public static void main(String[] args) throws IOException
 12     {
 13        try (Stream<String> lines = Files.lines(Paths.get("../words.txt")))
 14        {
 15           long count = lines
 16              .filter(w -> !w.endsWith("'s"))
 17              .filter(w -> hasFiveVowels(w))
 18              .count();
 19           System.out.println("Number of words with all five vowels: " + count);
 20        }
 21  
 22        try (Stream<String> lines = Files.lines(Paths.get("../words.txt")))
 23        {
 24           List<String> examples = lines 
 25              .filter(w -> !w.endsWith("'s"))
 26              .filter(w -> hasFiveVowels(w))
 27              .limit(20)
 28              .collect(Collectors.toList());
 29           System.out.println("20 words with all five vowels: " + examples);
 30        }
 31  
 32        try (Stream<String> lines = Files.lines(Paths.get("../words.txt")))
 33        {
 34           String shortest = lines
 35              .filter(w -> !w.endsWith("'s"))
 36              .filter(w -> hasFiveVowels(w))
 37              .min(Comparator.comparing(String::length))
 38              .orElse("");
 39           System.out.println("Shortest: " + shortest);
 40        }
 41        
 42        try (Stream<String> lines = Files.lines(Paths.get("../words.txt")))
 43        {
 44           List<String> allShortest = lines
 45              .filter(w -> !w.endsWith("'s"))
 46              .filter(w -> hasFiveVowels(w))
 47              .filter(w -> w.length() == 7)
 48              .collect(Collectors.toList());
 49           System.out.println("All of that length: " + allShortest);
 50        }
 51  
 52        try (Stream<String> lines = Files.lines(Paths.get("../words.txt")))
 53        {
 54           String longest = lines
 55              .filter(w -> !w.endsWith("'s"))
 56              .filter(w -> hasFiveVowels(w))
 57              .max(Comparator.comparing(String::length))
 58              .orElse("");
 59        
 60           System.out.println("Longest: " + longest);
 61        }
 62     }
 63     
 64     public static boolean hasFiveVowels(String word) 
 65     {
 66        return word.toLowerCase().codePoints()
 67           .filter(c -> c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u')
 68           .distinct()
 69           .count() == 5;
 70     }
 71  }
