1  import java.io.File;
  2  import java.io.IOException;
  3  import java.util.ArrayList;
  4  import java.util.List;
  5  import java.util.Optional;
  6  import java.util.Scanner;
  7  import java.util.stream.Stream;
  8  
  9  public class TerminalOperations
 10  {
 11     public static void main(String[] args) throws IOException
 12     {
 13        Scanner in = new Scanner(new File("../words.txt"));
 14        List<String> wordList = new ArrayList<>();
 15        while (in.hasNext()) { wordList.add(in.next()); }
 16        
 17        Stream<String> words = wordList.stream();
 18        
 19        String result = words.parallel()
 20           .filter(w -> w.length() > 10)
 21           .filter(w -> w.endsWith("y"))
 22           .findAny()
 23           .orElse("None found");
 24        // Note the use of findAny. Run this program several times
 25        // and check whether the result changes
 26        System.out.println("A word ending in y: " + result);
 27  
 28        words = wordList.stream();
 29        Optional<String> result2 = words.max((v, w) -> v.length() - w.length());
 30        System.out.print("The longest word: ");
 31        result2.ifPresent(w -> System.out.println(w));
 32  
 33        words = wordList.stream();
 34        boolean result3 = words.allMatch(w -> w.contains("e"));
 35        
 36        System.out.println("All words contain an e: " + result3);
 37     }
 38  }
