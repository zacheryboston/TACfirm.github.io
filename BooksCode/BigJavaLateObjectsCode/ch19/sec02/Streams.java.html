1  import java.io.IOException;
  2  import java.nio.file.Files;
  3  import java.nio.file.Paths;
  4  import java.util.ArrayList;
  5  import java.util.List;
  6  import java.util.stream.Stream;
  7  import java.util.stream.Collectors;
  8  
  9  public class Streams
 10  {
 11     public static void main(String[] args) throws IOException
 12     {
 13        Stream<String> words = Stream.of("Mary", "had", "a", "little", "lamb");
 14        show("A stream of individually specified words", words);
 15  
 16        Stream<Integer> digits = Stream.of(3, 1, 4, 1, 5, 9);
 17        show("A stream of individually specified digits", digits);
 18  
 19        Integer[] digitArray = { 3, 1, 4, 1, 5, 9 };
 20        digits = Stream.of(digitArray);
 21        show("A stream of digits from an array", digits);
 22  
 23        List<String> wordList = new ArrayList<>();
 24        // Populate wordList
 25        wordList.add("Hello");
 26        wordList.add("World");
 27        words = wordList.stream();
 28        show("A stream of words from a list", words);
 29        
 30        try (Stream<String> lines = Files.lines(Paths.get("../countries.txt")))
 31        {
 32           show("A stream of lines from a file", lines);
 33        }
 34     }
 35  
 36     /**
 37        Displays the first ten values in a stream, followed by ...
 38        if there are additional values.
 39        @param caption the caption preceding the values
 40        @param stream the stream to be displayed (which is consumed
 41        in the process)
 42     */
 43     public static <T> void show(String caption, Stream<T> stream)
 44     {
 45        final int SHOW_LIMIT = 10;
 46        List<T> values = stream.limit(SHOW_LIMIT + 1).collect(
 47           Collectors.toList());
 48        System.out.print(caption + ": ");
 49        for (int i = 0; i < Math.min(SHOW_LIMIT, values.size()); i++)
 50        {
 51           if (i > 0) { System.out.print(", "); }
 52           System.out.print(values.get(i));
 53        }
 54        if (values.size() > SHOW_LIMIT) { System.out.print(", ..."); }
 55        System.out.println();
 56     }
 57  }
