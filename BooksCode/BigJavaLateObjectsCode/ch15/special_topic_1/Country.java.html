1  import java.util.Objects;
  2  
  3  /**
  4     A country with a name and area.
  5  */
  6  public class Country
  7  {
  8     private String name;
  9     private double area;
 10  
 11     /**
 12        Constructs a country.
 13        @param aName the name of the country
 14        @param anArea the area of the country
 15     */
 16     public Country(String aName, double anArea) 
 17     { 
 18        name = aName;
 19        area = anArea; 
 20     }
 21  
 22     /**
 23        Gets the country name.
 24        @return the name
 25     */
 26     public String getName() 
 27     {
 28        return name;
 29     }
 30  
 31     /**
 32        Gets the area of the country.
 33        @return the area
 34     */
 35     public double getArea() 
 36     {
 37        return area;
 38     }
 39     
 40     public int hashCode()
 41     {
 42        return Objects.hash(name, area);
 43     }
 44  
 45     public boolean equals(Object otherObject)
 46     {  
 47        Country other = (Country) otherObject;
 48        return name.equals(other.name) 
 49           && area == other.area; 
 50     }
 51     
 52     public String toString()
 53     {
 54        return getClass().getName() + "[name=" + name 
 55           + ",area=" + area + "]";
 56     }
 57  }
