 36     public static void draw(Graphics graph)
 37     {
 38         // Your code goes here. To draw a line from (x1, y1) to (x2, y2), call
 39         // graph.drawLine(x1, y1, x2, y2);
 40         
 41         int x1 = 100;
 42         int y1 = 100;
 43         int length = 10;
 44         
 45         for (int i = 1; i <= 10; i++)
 46         {
 47             // Right
 48             int x2 = x1 + length;
 49             int y2 = y1;
 50             graph.drawLine(x1, y1, x2, y2);           
 51             x1 = x2;
 52             y1 = y2;
 53  
 54             // Up
 55             y2 = y1 - length;           
 56             graph.drawLine(x1, y1, x2, y2);
 57             x1 = x2;
 58             y1 = y2;
 59  
 60             length = length + 10;
 61  
 62             // Left
 63             x2 = x1 - length;
 64             graph.drawLine(x1, y1, x2, y2);
 65             x1 = x2;
 66             y1 = y2;
 67  
 68             // Down
 69             y2 = y1 + length;
 70             graph.drawLine(x1, y1, x2, y2);

 72             y1 = y2;
 73             
 74             length = length + 10;
 75         }
