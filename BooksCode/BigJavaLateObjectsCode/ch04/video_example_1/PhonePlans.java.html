1  import java.util.Scanner;
  2  
  3  public class PhonePlans
  4  {
  5     public static void main(String[] args)
  6     {
  7        Scanner in = new Scanner(System.in);
  8        System.out.print("Minutes used: ");
  9        int minutesUsed = in.nextInt();
 10        System.out.print("Messages used: ");
 11        int messagesUsed = in.nextInt();
 12        System.out.print("Megabyte used: ");
 13        int megabyteUsed = in.nextInt();    
 14  
 15        String cheapestPlanName = "";
 16        double cheapestPlanCost = 0;
 17        boolean done = false;
 18        while (!done)
 19        {
 20           System.out.print("Plan name (one word, Q to quit): ");
 21           String planName = in.next();
 22           if (planName.equals("Q")) 
 23           {
 24              done = true;
 25           }
 26           else
 27           {
 28              System.out.print("Phone cost: ");
 29              double phoneCost = in.nextDouble();
 30              System.out.print("Monthly fee: ");
 31              double monthlyFee = in.nextDouble();         
 32              System.out.print("Free minutes: ");
 33              int freeMinutes = in.nextInt();
 34              System.out.print("Free messages: ");
 35              int freeMessages = in.nextInt();
 36              System.out.print("Free megabyte: ");
 37              int freeMegabyte = in.nextInt();
 38              System.out.print("Fee for each additional minute: ");
 39              double additionalMinuteFee = in.nextDouble();
 40              System.out.print("Fee for each additional message: ");
 41              double additionalMessageFee = in.nextDouble();
 42              System.out.print("Fee for each additional megabyte: ");
 43              double additionalMegabyteFee = in.nextDouble();
 44  
 45              final int MONTHS_IN_CONTRACT = 24;
 46              double monthlyCost = phoneCost / MONTHS_IN_CONTRACT + monthlyFee;
 47  
 48              if (minutesUsed > freeMinutes)
 49              {
 50                 monthlyCost = monthlyCost 
 51                    + (minutesUsed - freeMinutes) * additionalMinuteFee;
 52              }
 53  
 54              if (messagesUsed > freeMessages)
 55              {
 56                 monthlyCost = monthlyCost 
 57                    + (messagesUsed - freeMessages) * additionalMessageFee;
 58              }
 59  
 60              if (megabyteUsed > freeMegabyte)
 61              {
 62                 monthlyCost = monthlyCost 
 63                    + (megabyteUsed - freeMegabyte) * additionalMegabyteFee;
 64              }
 65  
 66              if (cheapestPlanName.equals("") || cheapestPlanCost > monthlyCost)
 67              {
 68                 cheapestPlanName = planName;
 69                 cheapestPlanCost = monthlyCost;
 70              }
 71              System.out.println("Monthly cost of this plan: " + monthlyCost);
 72           }
 73        }
 74        System.out.println("Cheapest plan: " + cheapestPlanName);
 75        System.out.println("Monthly cost: " + cheapestPlanCost);
 76     }
 77  }
