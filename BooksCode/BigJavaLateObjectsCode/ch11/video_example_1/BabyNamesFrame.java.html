1  import java.awt.BorderLayout;
  2  import java.awt.GridLayout;
  3  import java.awt.event.ActionEvent;
  4  import java.awt.event.ActionListener;
  5  import java.io.File;
  6  import java.io.FileNotFoundException;
  7  import java.util.ArrayList;
  8  import java.util.Scanner;
  9  import javax.swing.ButtonGroup;
 10  import javax.swing.JButton;
 11  import javax.swing.JCheckBox;
 12  import javax.swing.JComboBox;
 13  import javax.swing.JFrame;
 14  import javax.swing.JLabel;
 15  import javax.swing.JOptionPane;
 16  import javax.swing.JPanel;
 17  import javax.swing.JRadioButton;
 18  import javax.swing.JScrollPane;
 19  import javax.swing.JTextField;
 20  import javax.swing.JTextArea;
 21  
 22  public class BabyNamesFrame extends JFrame
 23  {
 24     private ArrayList<String> boys;
 25     private ArrayList<String> girls;
 26  
 27     private JPanel choicesPanel;
 28     private JComboBox firstLetter;
 29     private JRadioButton boy;
 30     private JRadioButton girl;
 31     private JCheckBox isShort;
 32     private JCheckBox isMedium;
 33     private JCheckBox isLong;
 34     private JTextArea result;
 35     private ActionListener listener;
 36  
 37     private static final int SHORT_LENGTH = 5;
 38     private static final int LONG_LENGTH = 9;
 39     private static final int FRAME_WIDTH = 200;
 40     private static final int FRAME_HEIGHT = 500;
 41  
 42     public class UpdateListener implements ActionListener
 43     {
 44        public void actionPerformed(ActionEvent event)
 45        {
 46           ArrayList<String> names;
 47           if (boy.isSelected()) { names = boys; } else { names = girls; }
 48           result.setText("");
 49           for (String name : names)
 50           {
 51              if (matches(name))
 52              {
 53                 result.append(name + "\n");
 54              }
 55           }         
 56        }
 57  
 58        private boolean matches(String name)
 59        {
 60           String first = (String) firstLetter.getSelectedItem();
 61           int length = name.length();
 62           return name.startsWith(first)
 63              && (length <= SHORT_LENGTH && isShort.isSelected()
 64                 || SHORT_LENGTH < length && length < LONG_LENGTH 
 65                    && isMedium.isSelected()
 66                 || name.length() >= LONG_LENGTH && isLong.isSelected());
 67        }
 68     }
 69  
 70     public BabyNamesFrame()
 71     {
 72        choicesPanel = new JPanel();
 73        choicesPanel.setLayout(new GridLayout(3, 1));
 74        add(choicesPanel, BorderLayout.NORTH);
 75  
 76        listener = new UpdateListener();
 77        createFirstLetter();
 78        createBoyGirl();
 79        createLengths();
 80  
 81        result = new JTextArea();
 82        add(new JScrollPane(result), BorderLayout.CENTER);
 83  
 84        setSize(FRAME_WIDTH, FRAME_HEIGHT);
 85        
 86        readNames();
 87     }
 88  
 89     private void createFirstLetter()
 90     {
 91        JPanel row = new JPanel();
 92        choicesPanel.add(row);
 93        firstLetter = new JComboBox();
 94        for (char c = 'A'; c <= 'Z'; c++)
 95        {
 96           firstLetter.addItem("" + c);
 97        }
 98        firstLetter.addActionListener(listener);
 99        row.add(firstLetter);
100     }
101  
102     private void createBoyGirl()
103     {
104        JPanel row = new JPanel();
105        choicesPanel.add(row);
106        boy = new JRadioButton("Boy");
107        boy.addActionListener(listener);
108        girl = new JRadioButton("Girl");
109        girl.addActionListener(listener);
110        boy.setSelected(true);
111        row.add(boy);
112        row.add(girl);
113        ButtonGroup group = new ButtonGroup();
114        group.add(boy);
115        group.add(girl);
116     }
117  
118     private void createLengths()
119     {
120        JPanel row = new JPanel();
121        choicesPanel.add(row);
122        isShort = new JCheckBox("Short");
123        isShort.addActionListener(listener);
124        row.add(isShort);
125        isMedium = new JCheckBox("Medium");
126        isMedium.addActionListener(listener);
127        row.add(isMedium);
128        isLong = new JCheckBox("Long");
129        isLong.addActionListener(listener);
130        row.add(isLong);      
131     }
132  
133     private void readNames()
134     {
135        try
136        {
137           boys = new ArrayList<String>();
138           girls = new ArrayList<String>();
139           Scanner in = new Scanner(new File("babynames.txt"));
140           while (in.hasNext()) 
141           { 
142              boys.add(in.next()); 
143              girls.add(in.next()); 
144           }
145           in.close();
146        }
147        catch (FileNotFoundException ex)
148        {
149           JOptionPane.showMessageDialog(null, "babynames.txt file not found");
150        }     
151     }
152  }
