1  import java.awt.BorderLayout;
  2  import java.awt.GridLayout;
  3  import java.awt.event.ActionListener;
  4  import java.awt.event.ActionEvent;
  5  import javax.swing.JButton;
  6  import javax.swing.JFrame;
  7  import javax.swing.JPanel;
  8  import javax.swing.JTextField;
  9  
 10  /**
 11     This frame contains a panel that displays buttons
 12     for a calculator and a panel with a text fields to
 13     specify the result of calculation. The operator
 14     buttons don't do anything--see Worked Example 1
 15     for a working calculator.
 16  */
 17  public class CalculatorFrame extends JFrame
 18  {  
 19     private JTextField display;
 20  
 21     private static final int FRAME_WIDTH = 300;
 22     private static final int FRAME_HEIGHT = 300;
 23  
 24     public CalculatorFrame()
 25     {
 26        display = new JTextField("");
 27        display.setEditable(false);           
 28        add(display, BorderLayout.NORTH);
 29        createButtonPanel();
 30        setSize(FRAME_WIDTH, FRAME_HEIGHT);
 31     }
 32        
 33     /**
 34        Creates the button panel.
 35     */
 36     private void createButtonPanel()
 37     {      
 38        JPanel buttonPanel = new JPanel();
 39        buttonPanel.setLayout(new GridLayout(4, 4));
 40  
 41        buttonPanel.add(makeDigitButton("7"));
 42        buttonPanel.add(makeDigitButton("8"));
 43        buttonPanel.add(makeDigitButton("9"));
 44        buttonPanel.add(makeOperatorButton("/"));
 45        buttonPanel.add(makeDigitButton("4"));
 46        buttonPanel.add(makeDigitButton("5"));
 47        buttonPanel.add(makeDigitButton("6"));
 48        buttonPanel.add(makeOperatorButton("*"));
 49        buttonPanel.add(makeDigitButton("1"));
 50        buttonPanel.add(makeDigitButton("2"));
 51        buttonPanel.add(makeDigitButton("3"));
 52        buttonPanel.add(makeOperatorButton("-"));
 53        buttonPanel.add(makeDigitButton("0"));
 54        buttonPanel.add(makeDigitButton("."));
 55        buttonPanel.add(makeOperatorButton("="));
 56        buttonPanel.add(makeOperatorButton("+"));
 57  
 58        add(buttonPanel, BorderLayout.CENTER);
 59     }
 60     
 61     class DigitButtonListener implements ActionListener
 62     {
 63        private String digit;
 64        
 65        /**
 66           Constructs a listener whose actionPerformed method adds a digit
 67           to the display.
 68           @param aDigit the digit to add
 69        */
 70        public DigitButtonListener(String aDigit)
 71        {
 72           digit = aDigit;
 73        }
 74        
 75        public void actionPerformed(ActionEvent event)
 76        {  
 77           display.setText(display.getText() + digit);
 78        }        
 79     }
 80  
 81     /**
 82        Makes a button representing a digit of a calculator.
 83        @param digit the digit of the calculator
 84        @return the button of the calculator
 85     */
 86     public JButton makeDigitButton(String digit)
 87     {
 88        JButton button = new JButton(digit);
 89        
 90        ActionListener listener = new DigitButtonListener(digit);
 91        button.addActionListener(listener);  
 92        return button;    
 93     } 
 94  
 95     /**
 96        Makes a button representing an operator of a calculator.
 97        @param op the operator of the calculator
 98        @return button the button of the calcalator
 99     */
100     public JButton makeOperatorButton(String op)
101     {
102        JButton button = new JButton(op);
103        return button;    
104     }     
105  }
