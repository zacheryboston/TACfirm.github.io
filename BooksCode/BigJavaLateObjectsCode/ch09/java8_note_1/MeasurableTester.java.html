1  /**
  2     This program demonstrates a static method of the
  3     Measurable interface.
  4  */
  5  public class MeasurableTester
  6  {
  7     public static void main(String[] args)
  8     {
  9        // Calling the static average method
 10        // with an array of BankAccount objects
 11  
 12        Measurable[] accounts = new Measurable[3];
 13        accounts[0] = new BankAccount(0);
 14        accounts[1] = new BankAccount(10000);
 15        accounts[2] = new BankAccount(2000);
 16  
 17        double averageBalance = Measurable.average(accounts);
 18        System.out.println("Average balance: " + averageBalance);
 19        System.out.println("Expected: 4000");
 20  
 21        // Calling the static average method
 22        // with an array of Country objects
 23        
 24        Measurable[] countries = new Measurable[3];
 25        countries[0] = new Country("Uruguay", 176220);
 26        countries[1] = new Country("Thailand", 513120);
 27        countries[2] = new Country("Belgium", 30510);
 28  
 29        double averageArea = Measurable.average(countries);
 30        System.out.println("Average area: " + averageArea);
 31        System.out.println("Expected: 239950");
 32     }
 33  }
