1  /**
  2     A discussion board message.
  3  */
  4  public class Message
  5  {
  6     private User author;
  7     private String title;
  8     private String text;
  9  
 10     /**
 11        Constructs a message.
 12        @param author the message author
 13        @param title the message title
 14        @param text the message text
 15     */
 16     public Message(User author, String title, String text)
 17     {
 18        this.author = author;
 19        this.title = title;
 20        this.text = text;
 21     }
 22  
 23     /**
 24        Gets the message title.
 25        @return the title
 26     */
 27     public String getTitle()
 28     {
 29        return title;
 30     }
 31  
 32     /**
 33        Displays the message.
 34     */
 35     public void display()
 36     {
 37        System.out.println("From: " + author.getName());
 38        System.out.println("Title: " + getTitle());
 39        System.out.print(text);
 40     }
 41  }
