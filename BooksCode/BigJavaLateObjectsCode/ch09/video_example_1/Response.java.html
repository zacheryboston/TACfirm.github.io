1  /**
  2     A response to a discussion board message.
  3  */
  4  public class Response extends Message
  5  {
  6     private Message original;
  7  
  8     /**
  9        Constructs a response.
 10        @param author the message author
 11        @param original the message to which this is a response
 12        @param text the message text
 13     */
 14     public Response(User author, Message original, String text)
 15     {
 16        super(author, "", text);
 17        this.original = original;
 18     }
 19  
 20     public String getTitle()
 21     {
 22        return "Re: " + original.getTitle();
 23     }
 24  
 25     public void display()
 26     {
 27        super.display();
 28        System.out.println("---");
 29        original.display();
 30     }
 31  }
