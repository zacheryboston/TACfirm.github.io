1  /**
  2     A user of the discussion board.
  3  */
  4  public class User
  5  {
  6     private String name;
  7     private String password;
  8  
  9     /**
 10        Constructs a user.
 11        @param aName the user's name
 12        @param aPassword the user's password
 13     */
 14     public User(String aName, String aPassword)
 15     {
 16        name = aName;
 17        password = aPassword;
 18     }
 19  
 20     /**
 21        Gets the user's name.
 22        @return the name
 23     */
 24     public String getName()

 27     }
 28     
 29     /**
 30        Checks whether the password matches.
 31        @param aPassword the password to match
 32        @return true if the user's password matches aPassword
 33     */
 34     public boolean authenticate(String aPassword)
 35     {
 36        return password.equals(aPassword);
 37     }
 38  
 39     /**

 42     */

 49     }
