1  import java.util.ArrayList;
  2  import java.util.Scanner;
  3  
  4  /**
  5     A simulated discussion board.
  6  */
  7  public class DiscussionBoard
  8  {
  9     private ArrayList<User> users;
 10     private ArrayList<Message> messages;
 11     private User currentUser;
 12     private Message currentMessage;
 13     private Scanner in;
 14  
 15     public static void main(String[] args)
 16     {
 17        new DiscussionBoard().run();
 18     }
 19  
 20     /**
 21        Constructs the discussion board.
 22     */
 23     public DiscussionBoard()
 24     {
 25        in = new Scanner(System.in);
 26        users = new ArrayList<User>();
 27        messages = new ArrayList<Message>();
 28        User admin = new Instructor("admin", "secret");
 29        users.add(admin);
 30        Message welcome = new Message(admin, "Welcome", 
 31           "Welcome to the discussion board!\n");
 32        messages.add(welcome);
 33     }
 34  
 35     /**
 36        Runs the simulation.
 37     */
 38     public void run()
 39     {
 40        while (true)
 41        {
 42           if (currentUser == null)
 43           {
 44              login();
 45           }
 46           else
 47           {
 48              String userMenu = currentUser.getMenu(); 
 49              System.out.print("\n" + userMenu + "Command: ");
 50              String command = in.nextLine().toUpperCase();
 51              if (userMenu.contains(command + ")"))
 52              {
 53                 if (command.equals("S")) { showMessages(); }
 54                 else if (command.equals("N")) { newMessage(); }
 55                 else if (command.equals("R")) { replyToMessage(); }
 56                 else if (command.equals("L")) { logout(); }
 57                 else if (command.equals("P")) { purgeMessage(); }
 58                 else if (command.equals("A")) { addUser(); }
 59              }
 60           }
 61        }
 62     }
 63  
 64     /**
 65        Finds a user with a given name.
 66        @param username the name to search
 67        @return the matching user, or null if none found
 68     */
 69     public User findUser(String username)
 70     {
 71        boolean found = false;
 72        for (int i = 0; !found && i < users.size(); i++)
 73        {
 74           User current = users.get(i);
 75           if (current.getName().equals(username)) { return current; }
 76        }
 77        return null;
 78     }
 79     
 80     /**
 81        Executes the login command.
 82     */
 83     public void login()
 84     {
 85        System.out.print("User name: ");
 86        String username = in.nextLine();
 87        System.out.print("Password: ");
 88        String password = in.nextLine();
 89        User found = findUser(username);
 90        if (found != null && found.authenticate(password)) 
 91        {
 92           currentUser = found;
 93        }
 94        else
 95        {
 96           System.out.println("Authentication failed");
 97        }
 98     }
 99  
100     /**
101        Executes the add user command.
102     */
103     public void addUser()
104     {
105        System.out.print("User name: ");
106        String username = in.nextLine();
107        System.out.print("Password: ");
108        String password = in.nextLine();
109        if (findUser(username) == null) 
110        {         
111           users.add(new User(username, password));
112        }
113        else
114        {
115           System.out.println("User already exists");
116        }
117     }
118  
119     /**
120        Executes the logout command.
121     */
122     public void logout()
123     {
124        currentUser = null;
125     }
126  
127     /**
128        Shows all message titles and lets the user pick one for display.
129     */
130     public void showMessages()
131     {
132        for (int i = 0; i < messages.size(); i++)
133        {
134           Message msg = messages.get(i);
135           if (msg == currentMessage) System.out.print("*");
136           System.out.println((i + 1) + ") " + msg.getTitle());
137        }
138        System.out.print("Message number: ");
139        int number = in.nextInt(); in.nextLine(); 
140        currentMessage = messages.get(number - 1);
141        currentMessage.display();
142     }
143  
144     /**
145        Executes the new message command.
146     */
147     public void newMessage()
148     {
149        System.out.print("Title: ");
150        String title = in.nextLine();
151        String text = "";
152        System.out.println("Enter text, Q when done");
153        boolean done = false;
154        while (!done)
155        {
156           String line = in.nextLine();
157           if (line.toUpperCase().equals("Q")) { done = true; }
158           else { text = text + line + "\n"; }
159        }
160        currentMessage = new Message(currentUser, title, text);
161        messages.add(currentMessage);
162     }
163     
164     /**
165        Executes the reply to message command.
166     */
167     public void replyToMessage()
168     {
169        String text = "";
170        System.out.println("Enter reply, Q when done");
171        boolean done = false;
172        while (!done)
173        {
174           String line = in.nextLine();
175           if (line.toUpperCase().equals("Q")) { done = true; }
176           else { text = text + line + "\n"; }
177        }
178        currentMessage = new Response(currentUser, currentMessage, text);
179        messages.add(currentMessage);
180     }
181  
182     /**
183        Executes the purge message command.
184     */
185     public void purgeMessage()
186     {
187        messages.remove(currentMessage);
188     }
189  }
