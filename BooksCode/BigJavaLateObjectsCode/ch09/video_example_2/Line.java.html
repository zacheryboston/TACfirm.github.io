1  import java.awt.Graphics;
  2  
  3  public class Line extends Shape
  4  {
  5     private int x1;
  6     private int y1;
  7     private int x2;
  8     private int y2;
  9  
 10     public Line(int xfrom, int yfrom, int xto, int yto)
 11     {
 12        x1 = xfrom;
 13        y1 = yfrom;
 14        x2 = xto;
 15        y2 = yto;
 16     }
 17  
 18     /**
 19        Moves this shape by the given amount.
 20        @param dx the amount to move by in x-direction
 21        @param dy the amount to move by in y-direction
 22     */
 23     public void move(int dx, int dy) 
 24     {
 25        x1 = x1 + dx;
 26        x2 = x2 + dx;
 27        y1 = y1 + dy;
 28        y2 = y2 + dy;
 29     }
 30     
 31     public int getCenterX()
 32     {
 33        return (x1 + x2) / 2;
 34     }
 35  
 36     public int getCenterY()
 37     {
 38        return (y1 + y2) / 2;
 39     }
 40  
 41     /**
 42        Draws this shape
 43        g the Graphics object for the window in which the shape is drawn
 44     */
 45     public void draw(Graphics g) 
 46     {
 47        g.drawLine(x1, y1, x2, y2);
 48     }
 49  }
