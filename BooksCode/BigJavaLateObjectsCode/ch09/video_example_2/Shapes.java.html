1  import java.awt.Graphics;
  2  import java.awt.event.ActionEvent;
  3  import java.awt.event.ActionListener;
  4  import javax.swing.JButton;
  5  import javax.swing.JComponent;
  6  import javax.swing.JFrame;
  7  import javax.swing.JPanel;
  8  
  9  public class Shapes
 10  {   
 11     /** Generates random shapes
 12      * @param shapes the array to fill with shapes
 13      * @param xmax the maximum x-position
 14      * @param ymax the maximum y-position
 15      */ 
 16     public static void fillWithRandom(Shape[] shapes, int xmax, int ymax)
 17     {
 18        for (int i = 0; i < shapes.length; i++)
 19        {
 20           int type = (int) (Math.random() * 3);
 21           int x = (int) (Math.random() * xmax);
 22           int y = (int) (Math.random() * ymax);
 23           if (type == 0)
 24           {
 25              shapes[i] = new Circle(x, y, 10);
 26           }
 27           else if (type == 1)
 28           {
 29              shapes[i] = new Rectangle(x, y, 30, 20);
 30           }
 31           else
 32           {
 33              shapes[i] = new Line(x, y, x + 10, y + 20);
 34           }
 35        }
 36     }
 37     
 38     /**
 39      * Moves each shape by a small amount.
 40      * @param shapes the shapes
 41      */ 
 42     public static void moveBySmallAmount(Shape[] shapes)
 43     {
 44        final int X_OFFSET = 4;
 45        final int Y_OFFSET = 2;
 46        for (Shape s : shapes)
 47        {
 48           s.move(X_OFFSET, Y_OFFSET);
 49        }
 50     }
 51     
 52     /**
 53      * Lines up shapes in rows.
 54      * @param shapes the shapes
 55      * @param margin the right margin at which a new row should start
 56      */ 
 57     public static void lineUp(Shape[] shapes, int margin)
 58     {
 59        final int DISTANCE = 40;
 60        int x = DISTANCE / 2;
 61        int y = DISTANCE / 2;
 62        for (Shape s : shapes)
 63        {
 64           int dx = x - s.getCenterX();
 65           int dy = y - s.getCenterY();
 66           s.move(dx, dy);
 67           x = x + DISTANCE;
 68           if (x >= margin)
 69           {
 70              x = DISTANCE / 2;
 71              y = y + DISTANCE;
 72           }
 73        }
 74        
 75     }
 76     
 77     public static void main(String[] args)
 78     {
 79        final int XSIZE = 400;
 80        final int YSIZE = 400;
 81        final int NSHAPES = 30;
 82        final Shape[] shapes = new Shape[NSHAPES];
 83        fillWithRandom(shapes, XSIZE, YSIZE);
 84        
 85        final JFrame frame = new JFrame();
 86        frame.add(new JComponent()
 87        {
 88           public void paintComponent(Graphics g)
 89           {
 90              for (Shape s : shapes)
 91              {
 92                 s.draw(g);
 93              }
 94           }
 95        });
 96        
 97        JPanel panel = new JPanel();
 98        frame.add(panel, "North");
 99        JButton button1 = new JButton("Randomize");
100        panel.add(button1);
101        button1.addActionListener(new ActionListener()
102        {
103           public void actionPerformed(ActionEvent event)
104           {
105              fillWithRandom(shapes, XSIZE, YSIZE);
106              frame.repaint();
107           }
108        });
109  
110        JButton button2 = new JButton("Move");
111        panel.add(button2);
112        button2.addActionListener(new ActionListener()
113        {
114           public void actionPerformed(ActionEvent event)
115           {            
116              moveBySmallAmount(shapes);
117              frame.repaint();
118           }
119        });
120        
121        JButton button3 = new JButton("Line up");
122        panel.add(button3);
123        button3.addActionListener(new ActionListener()
124        {
125           public void actionPerformed(ActionEvent event)
126           {
127              lineUp(shapes, XSIZE);
128              frame.repaint();
129           }
130        });
131        
132  
133        frame.setSize(XSIZE, YSIZE);
134        frame.setVisible(true);      
135     }
136  }
