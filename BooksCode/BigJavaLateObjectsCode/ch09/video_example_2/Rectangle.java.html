1  import java.awt.Graphics;
  2  
  3  public class Rectangle extends Shape
  4  {
  5     private int xleft;
  6     private int ytop;
  7     private int width;
  8     private int height;
  9  
 10     public Rectangle(int x, int y, int w, int h)
 11     {
 12        xleft = x;
 13        ytop = y;
 14        width = w;
 15        height = h;
 16     }
 17  
 18     /**
 19        Moves this shape by the given amount.
 20        @param dx the amount to move by in x-direction
 21        @param dy the amount to move by in y-direction
 22     */
 23     public void move(int dx, int dy) 
 24     {
 25        xleft += dx;
 26        ytop += dy;
 27     }
 28   
 29     public int getCenterX()
 30     {
 31        return xleft + width / 2;
 32     }
 33     
 34     public int getCenterY()
 35     {
 36        return ytop + height / 2;
 37     }
 38     
 39     /**
 40        Draws this shape
 41        g the Graphics object for the window in which the shape is drawn
 42     */
 43     public void draw(Graphics g) 
 44     {
 45        g.drawRect(xleft, ytop, width, height);
 46     }
 47  }
