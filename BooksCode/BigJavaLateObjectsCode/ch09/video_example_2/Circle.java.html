1  import java.awt.Graphics;
  2  
  3  public class Circle extends Shape
  4  {
  5     private int centerX;
  6     private int centerY;
  7     private int radius;
  8  
  9     public Circle(int centerX, int centerY, int radius)
 10     {
 11        this.centerX = centerX;
 12        this.centerY = centerY;
 13        this.radius = radius;
 14     }
 15  
 16     /**
 17        Moves this shape by the given amount.
 18        @param dx the amount to move by in x-direction
 19        @param dy the amount to move by in y-direction
 20     */
 21     public void move(int dx, int dy) 
 22     {
 23        centerX += dx;
 24        centerY += dy;
 25     }
 26        
 27     public int getCenterX()
 28     {
 29        return centerX;
 30     }
 31  
 32     public int getCenterY()
 33     {
 34        return centerY;
 35     }
 36  
 37     /**
 38        Draws this shape
 39        g the Graphics object for the window in which the shape is drawn
 40     */
 41     public void draw(Graphics g) 
 42     {
 43        g.drawOval(centerX - radius, centerY - radius, 2 * radius, 2 * radius);
 44     }
 45  }
