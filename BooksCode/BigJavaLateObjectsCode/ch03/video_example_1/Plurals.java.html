 13        String plural;
 14        if (word.endsWith("ch") || word.endsWith("s")
 15           || word.endsWith("sh") || word.endsWith("x") 
 16           || word.endsWith("z"))
 17        {
 18           plural = word + "es";
 19        }
 20        else if (!(before == 'a' || before == 'e' || before == 'i'
 21              || before == 'o' || before == 'u') && last == 'y')
 22        {
 23           plural = word.substring(0, lastPos) + "ies";
 24        }
 25        else
 26        {
 27            plural = word + "s";
 28        }
 29        System.out.println("Plural: " + plural);
 30     }
