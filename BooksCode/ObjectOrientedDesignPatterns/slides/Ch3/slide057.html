JUnit

  • Each test case ends with assertTrue method
    (or another JUnit assertion method such as assertEquals)
  • Test framework catches assertion failures
public void testAdd()
{
   Day d1 = new Day(1970, 1, 1);
   int n = 1000;
   Day d2 = d1.addDays(n);
   assertTrue(d2.daysFrom(d1) == n);
}
